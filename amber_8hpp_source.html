149  // Reads in a "block" of data. Reading terminates on the first
150  // line that begins with a '%'.
151 
152  template<typename T>
153  std::vector<T> readBlock(const int field_width) {
154  std::vector<T> data;
155  while (reader.getNext()) {
156  std::string line = reader.line();
157  if (line[0] == '%') {
158  reader.push_back(line);
159  break;
160  }
161  std::istringstream iss(line);
162  T d;
163  while (iss >> std::setw(field_width) >> d)
164  data.push_back(d);
165  }
166 
167  return(data);
168  }
