37 string fullHelpMessage()
38  {
39 string s =
40  "Usage: rgyr SystemFile Trajectory selection min max num_bins skip [by-molecule]\n"
41  "\tby-molecule should be one if you want the selection\n"
42  "\tbroken up based on connectivity, and 0 or absent otherwise.\n"
43  "\n"
44  "\n"
45  "SYNOPSIS\n"
46  "\n"
47  "Read a trajectory and return a histogram of the radius of gyration of\n"
48  "a selection.\n"
49  "\n"
50  "DESCRIPTION\n"
51  "\n"
52  "This program computes the radius of gyration for a selection of atoms\n"
53  "for each frame of a trajectory. The default is to treat the selection\n"
54  "as one group, unless split-by-molecule is specified, which will split\n"
55  "the selection by connectivity.\n"
56  "\n"
57  "A minimum and maximum radius of gyration need to specified, as well as\n"
58  "the number of bins.\n"
59  "\n"
60  "EXAMPLE\n"
61  "\n"
62  "rgyr model-file traj.dcd 'resname==\"POPE\" 0 50 50 0 0\n"
63  "\n"
64  "In this instance, model-file is your input structure file and traj.dcd\n"
65  "is your trajectory. The selection string indicates that we are\n"
66  "computing the radius of gyration for all residues with the name POPE.\n"
67  "The minimum and maximum are set to 0 and 20, respectively, with bins of\n"
68  "one angstrom.\n"
69  "\n"
70  "Because split-by-molecule was not set, we are effectively computing\n"
71  "the radius of gyration of the entire set of POPE. This would\n"
72  "be useful if we were computing the radius of gyration of a micelle of\n"
73  "of POPE lipids. However, if we want the radius of gyration of individual\n"
74  "POPE lipids, we need to turn on split-by-molecule.\n"
75  "\n"
76  "rgyr model-file traj.dcd 'resname==\"POPE\" 0 20 20 0 1\n"
77  ;
78 return(s);
79  }

81 int main (int argc, char *argv[])
82 {
83 if ( (argc <= 1) ||
84  ( (argc >= 2) && (strncmp(argv[1], "-h", 2) == 0) ) ||
85  (argc < 8)
86  )
87  {
88  cerr << fullHelpMessage() << endl;
89  exit(-1);
90  }
91 
92 // Print the command line arguments
93 cout << "# " << invocationHeader(argc, argv) << endl;
94 
95 // Create the system and the trajectory file
96 AtomicGroup system = createSystem(argv[1]);
97 pTraj traj = createTrajectory(argv[2], system);
98 
99 char *selection = argv[3]; // String describing the first selection
100 
101 // Get the histogram parameters
102 greal hist_min = atof(argv[4]); // Lower edge of the histogram
103 greal hist_max = atof(argv[5]); // Upper edge of the histogram
104 int num_bins = atoi(argv[6]); // Number of bins in the histogram
105 
106 // Get the number of frames to discard as equilibration
107 int skip = atoi(argv[7]);
108 
109 int split_by_molecule = 0;
110 if (argc >= 9)
111  {
112  split_by_molecule = atoi(argv[8]);
113  }
114 
115 greal bin_width = (hist_max - hist_min)/num_bins;
116 
117 
118 
119 vector<AtomicGroup> molecules;
120 if (split_by_molecule)
121  {
122  molecules = system.splitByMolecule();
123  }
124 else
125  {
126  molecules.push_back(system);
127  }
128 
129 // Set up the selector to define the selected group
130 Parser parser(selection);
131 KernelSelector parsed_sel(parser.kernel());
132 
133 
134 // Loop over the molecules and add them to selection
135 vector<AtomicGroup> molecule_groups;
136 vector<AtomicGroup>::iterator m;
137 for (m=molecules.begin(); m!=molecules.end(); m++)
138  {
139  AtomicGroup tmp = m->select(parsed_sel);
140  if (tmp.size() > 0)
141  {
142  molecule_groups.push_back(tmp);
143  }
144  }
145 
146 
147 // Skip the initial frames as equilibration
148 if (skip > 0)
149 {
150  traj->readFrame(skip - 1);
151 }
152 
153 // read the initial coordinates into the system
154 traj->updateGroupCoords(system);
155 
156 // Create the histogram and zero it out
157 vector<greal> hist;
158 hist.reserve(num_bins);
159 hist.insert(hist.begin(), num_bins, 0.0);
160 
161 // loop over the frames of the trajectory
162 int frame = 0;
163 int count = 0;
164 while (traj->readFrame())
165  {
166  // update coordinates and periodic box
167  traj->updateGroupCoords(system);
168 
169  vector<AtomicGroup>::iterator m;
170  for (m=molecule_groups.begin(); m!=molecule_groups.end(); m++)
171  {
172  greal rad = m->radiusOfGyration();
173  if ( (rad >=hist_min) && (rad <hist_max) )
174  {
175  int bin = int((rad-hist_min)/bin_width);
176  hist[bin]++;
177  count++;
178  }
179  }
180  frame++;
181  }
182 
183 
184 // Output the results
185 cout << "# Rgyr\tProb\tCum" << endl;
186 greal cum = 0.0;
187 for (int i = 0; i < num_bins; i++)
188  {
189  greal d = bin_width*(i + 0.5) + hist_min;
190 
191 
192  greal prob = hist[i]/ count;
193  cum += prob;
194  cout << d << "\t"
195  << prob << "\t"
196  << cum << endl;
197 
198  }
199 }
