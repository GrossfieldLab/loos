38  bool LineReader::getNext() {
39  if (! _lines.empty()) {
40  _current_line = _lines.back();
41  _lines.pop_back();
42  } else if (_is->eof())
43  return(false);
44  else {
45  while (getline(*_is, _current_line)) {
46  ++_lineno;
47  stripComment(_current_line);
48  stripLeadingWhitespace(_current_line);
49  if (! skipLine(_current_line) )
50  break;
51  }
52  }
53 
54  checkState();
55  return( _is->good() );
56  }
