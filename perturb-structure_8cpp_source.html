94  bool postConditions(po::variables_map& map) {

116 int main(int argc, char *argv[]) {
117  string hdr = invocationHeader(argc, argv);
118  opts::BasicOptions* bopts = new opts::BasicOptions;
119  opts::BasicSelection* sopts = new opts::BasicSelection;
120  opts::ModelWithCoords* mwcopts = new opts::ModelWithCoords;
121  ToolOptions* topts = new ToolOptions;
122 
123  opts::AggregateOptions options;
124  options.add(bopts).add(sopts).add(mwcopts).add(topts);
125  if (!options.parse(argc, argv))
126  exit(-1);
127 
128  AtomicGroup model = mwcopts->model;
129  AtomicGroup subset = selectAtoms(model, sopts->selection);
130 
131  subset.perturbCoords(topts->magnitude);
132  PDB pdb = PDB::fromAtomicGroup(model);
133  pdb.remarks().add(hdr);
134 
135  cout << pdb;
136 }
