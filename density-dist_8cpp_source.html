119  "\n"
120  " Compute the electron, mass, or charge density for the system and\n"
121  " its components along the z-axis.\n"
122  "\n"
123  " DESCRIPTION\n"
124  "\n"
125  " The purpose of this tool is to computed the distribution of a system\n"
126  " along the z-axis. This is most useful for membrane systems, where \n"
127  " the data provided is analogous to that from x-ray or neutron \n"
128  " scattering. By default the program computes the total distribution,\n"

138  " FOO is either mass in AMU or charge/electron density in electrons.\n"

140  " Options\n"

209 int main(int argc, char *argv[]) {
210 
211  string hdr = invocationHeader(argc, argv);
212 
213  // Options handling...
214  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
215  opts::OutputPrefix* popts = new opts::OutputPrefix;
216  opts::BasicTrajectory* tropts = new opts::BasicTrajectory;
217 
218  // The required options could have been folded into ToolOptions, but
219  // using RequiredArguments saves us from having to handle the help()
220  // and print() methods...
221  opts::RequiredArguments* ropts = new opts::RequiredArguments;
222  ropts->addArgument("minz", "min-z");
223  ropts->addArgument("maxz", "max-z");
224  ropts->addArgument("nbins", "number-of-bins");
225  ToolOptions* topts = new ToolOptions;
226 
227  opts::AggregateOptions options;
228  options.add(bopts).add(popts).add(tropts).add(ropts).add(topts);
229  if (!options.parse(argc, argv)) {
230  cerr << endl;
231  cerr << "**Important note**\nYou must place '--' on the command line AFTER\n";
232  cerr << "the options if you are going to use a negative Z argument, i.e.\n";
233  cerr << "density-dist --type charge -- foo.pdb foo.dcd -40 40 40\n";
234  exit(-1);
235  }
236 
237  double min_z = parseStringAs<double>(ropts->value("minz"));
238  double max_z = parseStringAs<double>(ropts->value("maxz"));
239  uint nbins = parseStringAs<uint>(ropts->value("nbins"));
240  vector<string> selections = topts->selections;
241 
242  AtomicGroup system = tropts->model;
243  pTraj traj = tropts->trajectory;
244  // End of options
245  cout << "# " << hdr << endl;
246 
247  // density from each selection
248  vector<AtomicGroup> subsets;
249  for (vector<string>::iterator i = selections.begin(); i != selections.end(); ++i)
250  subsets.push_back(selectAtoms(system, *i));
251 
252  // Verify properties...
253  if (topts->calc_type != NUMBER) {
254  for (vector<AtomicGroup>::iterator i = subsets.begin(); i != subsets.end(); ++i) {
255  bool ok;
256  switch(topts->calc_type) {
257  case CHARGE: ok = i->allHaveProperty(Atom::chargebit); break;
258  case ELECTRON: ok = (i->allHaveProperty(Atom::anumbit) && i->allHaveProperty(Atom::chargebit)); break;
259  case MASS: ok = i->allHaveProperty(Atom::massbit); break;
260  default: cerr << "Internal Error- unknown calculation type\n"; exit(-1);
261  }
262  if (!ok) {
263  cerr << "***WARNING***\n";
264  cerr << "The system is missing properties required for calculation type.\n";
265  cerr << "Default values will be used where possible.\n";
266  cerr << "This may result in incorrect or absurd values.\n";
267  break;
268  }
269  }
270  }
271 
272  double bin_width = (max_z - min_z) / nbins;
273 
274  // Create and zero out the total charge_distribution
275  vector< vector<double> > dists(subsets.size(), vector<double>(nbins, 0.0));
276  // If we do windowed time series too, we'll need to zero out as we go, which
277  // means we'll need separate storage for the sum
278  vector< vector<double> > cum_dists(subsets.size(), vector<double>(nbins, 0.0));
279 
280  // Note: the equillibration frames are already skipped by opts::BasicTrajectory
281 
282  // loop over the remaining frames
283  uint frame = 0;
284  while (traj->readFrame()) {
285  // update coordinates
286  traj->updateGroupCoords(system);
287 
288  // Compute the bin volume for normalization purposes
289  GCoord box = system.periodicBox();
290  double bin_volume = bin_width * box.x() * box.y();
291 
292  // Loop over the subsets and compute charge distribution.
293  // (first set is all atoms)
294  for (uint i=0; i<subsets.size(); i++) {
295  double weight;
296  for (AtomicGroup::iterator atom = subsets[i].begin();
297  atom != subsets[i].end();
298  ++atom) {
299  switch(topts->calc_type) {
300  case CHARGE: weight = (*atom)->charge(); break;
301  case MASS: weight = (*atom)->mass(); break;
302  case ELECTRON: weight = (*atom)->atomic_number() - (*atom)->charge(); break;
303  case NUMBER: weight = 1.0; break;
304  default:
305  cerr << "ERROR- unknown calculation type\n";
306  exit(-1);
307  }
308  double z = (*atom)->coords().z();
309  if (topts->symmetrize)
310  z = abs(z);
311 
312  if ( (z > min_z) && (z < max_z) ) {
313  uint bin = (int)( (z - min_z) / bin_width );
314  dists[i][bin] += weight/bin_volume;
315  }
316  }
317  }
318  ++frame;
319 
320  // If windowed time series were requested, output them here
321  if (topts->window && (frame % topts->window == 0)) {
322  uint output_val = frame / topts->window;
323 
324  // build the output file name
325  char piece[128];
326  snprintf(piece, 128, "_%d.dat", output_val);
327  string file_name = popts->prefix + string(piece);
328  ofstream outfile(file_name.c_str());
329  if (!outfile.is_open() ) {
330  throw(runtime_error("couldn't open output file"));
331  }
332 
333  // write the header
334  outfile << "# Z\t";
335  outfile << "AllAtoms\t";
336  for (uint i=1; i<subsets.size(); i++) {
337  outfile << " Set(" << i <<") ";
338  }
339  outfile << endl;
340 
341  // Normalize by the number of frames and
342  // output the average density
343  for (uint i=0; i<nbins; i++) {
344  double z = (i+0.5)*bin_width + min_z;
345  outfile << z << "\t";
346  for (unsigned int j=0; j<subsets.size(); j++) {
347  double c = dists[j][i] / topts->window;
348  outfile << c << "\t";
349  }
350  outfile << endl;
351  }
352 
353  // Add the windowed density to the cumulative density and
354  // zero out the distributions
355 
356  for (uint i=0; i<dists.size(); i++) {
357  for (uint j=0; j<nbins; j++) {
358  cum_dists[i][j] += dists[i][j];
359  }
360 
361  // zero out the windowed dist
362  dists[i].assign(dists[i].size(), 0.0);
363  }
364 
365  }
366 
367 
368  }
369 
370  // Normalize by the number of frames and output the average charge density
371  cout << "# Z\tAllAtoms";
372  for (uint i=1; i<subsets.size(); i++) {
373  cout << " Set(" << i <<") ";
374  }
375  cout << endl;
376 
377  // If we didn't output windowed time series, we never used cum_dists, so we need
378  // to copy it over here
379  // If we did, we may have some data in dists that hasn't been added to cum_dists yet
380  if (!topts->window) {
381  cum_dists = dists;
382  }
383  else if (frame % topts->window != 0){
384  for (uint i=0; i<dists.size(); i++) {
385  for (uint j=0; j<nbins; j++) {
386  cum_dists[i][j] += dists[i][j];
387  }
388  }
389  }
390 
391  for (uint i=0; i<nbins; i++) {
392  double z = (i+0.5)*bin_width + min_z;
393  cout << z << "\t";
394  for (uint j=0; j<subsets.size(); j++) {
395  double c = cum_dists[j][i] / frame;
396 
397  cout << c << "\t";
398  }
399  cout << endl;
400  }
401 
402 
403 
404 }
