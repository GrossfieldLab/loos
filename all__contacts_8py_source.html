111 system = loos.createSystem(args.system_file)
112 all_trajs = []
113 out_names = []
114 num_trajs = len(args.traj_files)
115 for t in args.traj_files:
116  traj = loos.pyloos.Trajectory(t, system)
117  all_trajs.append(traj)
118  if (num_trajs > 1) and args.individual:
119  t_base = basename(t)
120  core, ext = splitext(t_base)
121  out_names.append(core + ".dat")
122 
123 if args.no_hydrogens:
124  no_hydrogens = loos.selectAtoms(system, "!hydrogen")
125  target = loos.selectAtoms(no_hydrogens, args.selection)
126 else:
127  target = loos.selectAtoms(system, args.selection)
128 
129 residues = target.splitByResidue()
130 # now remove the backbone -- doing before the split loses the glycines
131 if args.no_backbone:
132  residues = list([loos.selectAtoms(r, "!backbone") for r in residues])
133 
134 frac_contacts = numpy.zeros([len(residues), len(residues), num_trajs],
135  numpy.float)
136 
137 
138 for traj_id in range(num_trajs):
139  traj = all_trajs[traj_id]
140  for frame in traj:
141  for i in range(len(residues)):
142  for j in range(i+1, len(residues)):
143  if residues[i].contactWith(args.cutoff, residues[j]):
144  frac_contacts[i, j, traj_id] += 1.0
145  frac_contacts[j, i, traj_id] += 1.0
146  frac_contacts[:, :, traj_id] /= len(traj)
147  if (num_trajs > 1) and args.individual:
148  numpy.savetxt(out_names[traj_id], frac_contacts[:, :, traj_id],
149  header=header)
150 
151 average = numpy.add.reduce(frac_contacts, axis=2)
152 average /= len(args.traj_files)
153 
154 numpy.savetxt(args.out_file, average, header=header)
