62 void fullHelpMessage() {
63  //string msg =
64  cout << "\n"
65  "SYNOPSIS\n"
66  "\n"
67  "Calculate the covariance overlap between two eigendecompositions\n"
68  "\n"
69  "DESCRIPTION\n"
70  "\n"
71  "Coverlap compares two sets of eigendompositions (eigenvalues and\n"
72  "eigenvectors) using the metric described by Hess. 2002, Phys Rev E\n"
73  "(10.1103/PhysRevE.65.031910).\n"
74  "\n"
75  "In simplest terms this is a cumulative dot product between two\n"
76  "eigenvectors that is weighted relatively by each associated eigenvalue.\n"
77  "The nominclature of the input is broken down as:\n"
78  "\tls - left side eigenvalue file (filename_s.asc if made with loos)\n"
79  "\tlU - left side eigenvector file (filename_U.asc if made with loos)\n"
80  "\trs - right side eigenvalue file (filename_s.asc if made with loos)\n"
81  "\trU - right side eigenvector file (filename_U.asc if made with loos)\n"
82  "\t\n"
83  "\n"
84  "The results commonly compared are simulation PCA's and/or ENM-NMA's.\n"
85  "ENM eigenpairs are handled differently from PCA eigenpairs, you must specify\n"
86  "which sides are ENM results. This is done with the -e1 for ENM results on\n"
87  "the left side and -E1 for ENM results on the right side. Additionally, PCA \n"
88  "eigenpairs can be real eigenpairs or they can come from an SVD, in which \n"
89  "case the 'eigenvalues' must be squared. This is done with -s1 to square the\n"
90  "left side eigenvalues and -S1 to square the right side eigenvalues.\n"
91  "Finally, when comparing ENM and PCA you will probably want to scale the\n"
92  "eigenvalues such that the total power on each side are comparable.\n"
93  "The -p1 option does this.\n"
94  "\n"
95  "In addition a non-weighted overlap between eigenvectors can be computed.\n"
96  "Since the eigenvectors are complete basis sets this will always sum to 1\n"
97  "Therefore, a subset of the modes should be used. The number of modes to\n"
98  "use with this calculation is specified with the -u## argument.\n"
99  "\n"
100  "Another option is to compute the Z-score of a covariance overlap result.\n"
101  "This is done using a boot-strap approach to scramble eigenpairs and compute\n"
102  "a new overlap. this is done -z=### times, which is used to calculate the\n"
103  "average and standard deviation for the Z-score. For more details see:\n"
104  "\n"
105  " Romo, T. D., and Grossfield, A. Block covariance overlap method and \n"
106  " convergence in molecular dynamics simulation, J. Chem. Theor. Comput.,\n"
107  " 2011, 7, 2464-2472\n"
108  "\n"
109  " Leioatts, N., Romo, T. D., and Grossfield, A., Elastic network models \n"
110  " are robust to variations in formalism, J. Chem. Theor. Comput., 2012, \n"
111  " 8, 2424-2434\n"
112  "\n"
113  "\n"
114  "EXAMPLES\n"
115  "\n"
116  "\tcoverlap -e1 -S1 -p1 -u50 anm_s.asc anm_U.asc pca_s.asc pca_U.asc\n"
117  "This computes the covariance overlap between an ANM result (the left side)\n"
118  "and a PCA (the right side) that came from an SVD (right side). On the right\n"
119  "the singular values are squared (to make them eigenvalues) and they are\n"
120  "scaled to match the ANM eigenvalues (-p1). Finally, a subspace overlap using\n"
121  "the first 50 modes is also computed.\n"
122  "\n"
123  "\tcoverlap -e1 -p1 -u50 anm_s.asc anm_U.asc pca_s.asc pca_U.asc\n"
124  "The same as the above example, but here the PCA came from an eigendecomp,\n"
125  "so the eigenvalues used are real eigenvalues and do not need to be squared.\n"
126  "\n"
127  "\tcoverlap -e1 -E1 -u25 anm_s.asc anm_U.asc vsa_s.asc vsa_U.asc\n"
128  "This computes the covariance overlap between an ANM and a VSA model.\n"
129  "No scaling is applied to either side. The subspace overlap using the\n"
130  "first 25 modes is also computed.\n"

132  "\tcoverlap -e1 -E1 -u25 -k 1.234 anm_s.asc anm_U.asc vsa_s.asc vsa_U.asc\n"

134  "ANM eigenvalues.\n"

201 DMDuple transformENM(const DoubleMatrix& S, const DoubleMatrix& U, const uint nmodes) {
202  DoubleMatrix SS(nmodes, 1);
203  DoubleMatrix UU(U.rows(), nmodes);
204 
205  for (uint i=skip; i<nmodes+skip; ++i) {
206  SS[i-skip] = 1.0 / S[i];
207  for (uint j=0; j<U.rows(); ++j)
208  UU(j,i-skip) = U(j,i);
209  }
210 
211  return(DMDuple(SS,UU));
212 }

215 DMDuple firstColumns(const DoubleMatrix& S, const DoubleMatrix& U, const uint nmodes) {
216  DoubleMatrix SS(nmodes, 1);
217  DoubleMatrix UU(U.rows(), nmodes);
218 
219  for (uint i=0; i<nmodes; ++i) {
220  SS[i] = i < S.rows() ? S[i] : 0.0;
221 
222  for (uint j=0; j<U.rows(); ++j)
223  UU(j, i) = U(j, i);
224  }
225 
226  return(DMDuple(SS,UU));
227 }

231 DoubleMatrix scalePower(const DoubleMatrix& A, const DoubleMatrix& B) {
232 
233  double sumB = 0.0;
234  double sumA = 0.0;
235  for (uint j=0; j<B.rows(); ++j) {
236  sumB += B[j];
237  sumA += A[j];
238  }
239 
240  double scale = sumA / sumB;
241  cerr << "Scale factor = " << scale << endl;
242  DoubleMatrix E(B.rows(), 1);
243  for (uint j=0; j<B.rows(); ++j)
244  E[j] = B[j] * scale;
245 
246  return(E);
247 }

271 
272  if (number_of_modes == 0) {
273  number_of_modes = lS.rows() > rS.rows() ? lS.rows() : rS.rows();
274  if (left_is_enm || right_is_enm)
275  number_of_modes -= skip;
276  }
277 
278

292  lSS = boost::get<0>(res);

301  rUU = boost::get<1>(res);
302  } else {
303 
304  DMDuple res = firstColumns(rS, rU, number_of_modes);
305  rSS = boost::get<0>(res);
306  rUU = boost::get<1>(res);
307  }
308 
309 
310 
311  if (square_left)
312  for (uint j=0; j<lSS.rows(); ++j)
313  lSS[j] *= lSS[j];
314 
315  if (square_right)
316  for (uint j=0; j<rSS.rows(); ++j)
317  rSS[j] *= rSS[j];
318 
319 
320  for (uint j=0; j<rSS.rows(); ++j) {
321  rSS[j] *= rscale;
322  lSS[j] *= lscale;
323  }
324 
325  if (scale_power)
326  rSS = scalePower(lSS, rSS);
327 
328 
329  cout << "Covariance Modes: " << number_of_modes << endl;
330  if (ntries == 0) {
331  double overlap = covarianceOverlap(lSS, lUU, rSS, rUU);
332  cout << "Covariance overlap: " << overlap << endl;
333  } else {
334  if (seed == 0)
335  randomSeedRNG();
336  else
337  rng_singleton().seed(seed);
338 
339  boost::tuple<double,double,double> overlap = zCovarianceOverlap(lSS, lUU, rSS, rUU, ntries);
340  cout << "Covariance overlap: " << boost::get<1>(overlap) << endl;
341  cout << "Z-score: " << boost::get<0>(overlap) << endl;
342  }
343 
344  double subover = subspaceOverlap(lUU, rUU, subspace_size);
345  cout << "Subspace Modes: " << subspace_size << endl;
346  cout << "Subspace overlap: " << subover << endl;
