57 string fullHelpMessage(void) {
58  string msg =
59  "\n"
60  "SYNOPSIS\n"
61  "Aligns the structures in a trajectory\n"
62  "\n"
63  "DESCRIPTION\n"
64  "\taligner can align a trajectory two different ways: an optimal alignment of all structures,\n"
65  "or aligning all structures to a reference structure. The optimal alignment uses an iterative\n"
66  "algorithm (described in Grossfield, et al. Proteins 67, 31â€“40 (2007)). The latter method\n"
67  "uses a Kabsch least-squares algorithm to superimpose each frame of the trajectory onto the\n"
68  "reference structure.\n"
69  "\taligner can take multiple selections to govern what is aligned. The --align option\n"
70  "determines what subset of atoms from the trajectory is used in computing the alignment\n"
71  "transformation for each frame. The --transform option determines which atoms from the\n"
72  "trajectory the alignment transformation is applied to. Only these atoms are written out\n"
73  "to the DCD file generated by aligner. Note that the default --transform selection is 'all',\n"
74  "so if you want to align alpha-carbons and only output alpha-carbons, you will need to provide\n"
75  "the same selection (i.e. \"name == 'CA'\") to both options.\n"
76  "\tWhen aligning to a reference structure, the selection given in --align is used for\n"
77  "both the trajectory and the reference structure. Sometimes, however, the selection expression\n"
78  "for the trajectory does not match the reference structure (e.g. aligning a trajectory to a\n"
79  "crystal structure). In this case, use the --refsel option to specify a different selection\n"
80  "expression for the reference structure. Note that the atoms must be in the same order as\n"
81  "the ones picked by the trajectory --align (i.e. the first atom from the reference subset\n"
82  "must match the first atom in the trajectory subset, etc).\n"
83  "\tThe --xyonly and --noztrans options may be of interest for membrane systems.\n"
84  "The --xyonly option will translate the alignment subset, but will only rotate about\n"
85  "the Z-axis (i.e. within the x,y-plane). The --noztrans option will remove any Z-translation\n"
86  "in the subset after alignment. For example, aligning with --xyonly turned on will orient\n"
87  "the subset such that it is aligned within the x,y-plane, but the tilt relative to the x,y-plane\n"
88  "is preserved. However, the centroid of the selection will always be centered. If the --noztrans\n"
89  "option is turned on, then the selection will be centered in x and y, but the z-coordinates\n"
90  "will be preserved.\n"
91  "\n"
92  "\n"
93  "EXAMPLES\n"
94  "\n"
95  "\taligner --prefix b2ar-aligned b2ar.pdb b2ar.dcd\n"
96  "Aligns the trajectory based on the default selection (alpha-carbons). All atoms are transformed\n"
97  "and the model is written to b2ar-aligned.pdb and the aligned trajectory is written to\n"
98  "b2ar-aligned.dcd\n"
99  "\n"
100  "\taligner --prefix aligned --transform 'name =~ \"^(C|O|N|CA)$\"' model.pdb traj.dcd\n"
101  "Aligns the trajectory based on the default selection (alpha-carbons). Only backbone atoms are\n"
102  "transformed. Creates aligned.pdb and aligned.dcd\n"
103  "\n"
104  "\taligner --prefix aligned --align 'segid == \"HEME\"' \\\n"
105  "\t --transform 'segid == \"PROT\" || segid == \"HEME\"' model.pdb traj.dcd\n"
106  "Aligns the trajectory based on atoms with segid 'HEME'. Only transforms protein and heme atoms\n"
107  "\n"
108  "\taligner --prefix aligned --reference xtal.pdb model.pdb traj.dcd\n"
109  "Aligns using alpha-carbons, transforming all atoms, but align the trajectory to the structure\n"
110  "in xtal.pdb\n"
111  "\n"
112  "\taligner --prefix aligned --reference xtal.pdb \\\n"
113  "\t --refsel 'resid >= 30 && resid <= 60 && name == \"CA\"' \\\n"
114  "\t --align 'resid >= 10 && resid <= 40 && name == \"CA\"' model.pdb traj.dcd\n"
115  "Aligns against a reference structure. Uses alpha-carbons from residues 30-60 in the reference\n"

121  "\tSelecting too few atoms to align may result in a poor alignment.\n";

171 void zapZ(vMatrix& M) {
172  uint n = M[0].size();
173  for (uint i=0; i<M.size(); ++i)
174  for (uint j=2; j<n; j += 3)
175  M[i][j] = 0.0;
176 }

179 void zapZ(AtomicGroup& model) {
180  for (AtomicGroup::iterator i = model.begin(); i != model.end(); ++i)
181  (*i)->coords().z(0.0);
182 }

199 int main(int argc, char *argv[]) {
200 
201  // Parse command-line options, cache invocation header for later use...
202  string header = invocationHeader(argc, argv);
203  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
204  opts::OutputPrefix* prefopts = new opts::OutputPrefix;
205  opts::TrajectoryWithFrameIndices* tropts = new opts::TrajectoryWithFrameIndices;
206  opts::OutputTrajectoryTypeOptions *otopts = new opts::OutputTrajectoryTypeOptions;
207  ToolOptions* topts = new ToolOptions;
208 
209  opts::AggregateOptions options;
210  options.add(bopts).add(prefopts).add(tropts).add(otopts).add(topts);
211  if (!options.parse(argc, argv))
212  exit(-1);
213 
214  verbosity = bopts->verbosity;
215 
216  // Read the inputs...
217  AtomicGroup model = tropts->model;
218  pTraj traj = tropts->trajectory;
219 
220  // Get the selections (subsets) to operate over
221  AtomicGroup align_sub = selectAtoms(model, topts->alignment_string);
222  if (align_sub.size() < min_align_selection_warning)
223  cerr << "Warning- selecting fewer than " << min_align_selection_warning << " atoms with --align may result in a poor quality alignment.\n";
224 
225 
226  AtomicGroup applyto_sub = selectAtoms(model, topts->transform_string);
227 
228 
229 
230  // Now do the alignin'...
231  vector<uint> indices = tropts->frameList();
232  unsigned int nframes = indices.size();
233 
234  if (topts->reference_name.empty()) {
235 
236  // estimate memory requirements...
237  long used_memory = 3 * align_sub.size() * indices.size() * sizeof(double);
238  if (used_memory * 100l / availableMemory() > memory_threshold) {
239  cerr << boost::format("Warning- estimating that memory used for aligning is greater than %d%% of system memory.\n") % memory_threshold;
240  cerr << " Consider subsampling the trajectory or using a smaller alignment selection.\n";
241  }
242 
243  if (verbosity)
244  cerr << "Reading coordinates...\n";
245  vMatrix coords = readCoords(align_sub, traj, indices, verbosity);
246  if (topts->xy_only)
247  zapZ(coords);
248 
249  boost::tuple<vector<XForm>,greal, int> res = iterativeAlignment(coords, topts->alignment_tol, topts->maxiter);
250  greal final_rmsd = boost::get<1>(res);
251  cerr << "Final RMSD between average structures is " << final_rmsd << endl;
252  cerr << "Total iters = " << boost::get<2>(res) << endl;
253 
254  vector<XForm> xforms = boost::get<0>(res);
255 
256  // Zapping z-coords will leave Z part of xform 0, so must fix...
257  if (topts->xy_only) {
258  for (vector<XForm>::iterator i = xforms.begin(); i != xforms.end(); ++i) {
259  GMatrix M = i->current();
260  M(2,2) = 1.0;
261  i->load(M);
262  }
263  }
264 
265  // Setup for writing Trajectory
266  pTrajectoryWriter outtraj = otopts->createTrajectory(prefopts->prefix);
267  outtraj->setComments(header);
268 
269  // Now apply the alignment transformations to the requested subsets
270  for (unsigned int i = 0; i<nframes; i++) {
271  traj->readFrame(indices[i]);
272  traj->updateGroupCoords(model);
273  GCoord original_center;
274  if (topts->no_ztrans)
275  original_center = applyto_sub.centroid();
276 
277  model.applyTransform(xforms[i]);
278  if (topts->no_ztrans) {
279  GCoord new_center = applyto_sub.centroid();
280  double dz = original_center.z() - new_center.z();
281  GCoord shift(0.0, 0.0, dz);
282  applyto_sub.translate(shift);
283  }
284 
285  outtraj->writeFrame(applyto_sub);
286 
287  if (i == 0)
288  savePDB(prefopts->prefix + ".pdb", header, applyto_sub);
289  }
290 
291  } else { // else, aligning to reference structure (i.e. non-iterative)
292 
293  AtomicGroup reference = createSystem(topts->reference_name);
294 
295  string refsel = topts->reference_sel.empty() ? topts->alignment_string : topts->reference_sel;
296  AtomicGroup refsub = selectAtoms(reference, refsel);
297 
298  if (refsub.size() != align_sub.size()) {
299  cerr << boost::format("ERROR- alignment subset has %d atoms but reference subset has %d. They must match.\n") % align_sub.size() % refsub.size();
300  exit(-10);
301  }
302 
303  if (topts->xy_only)
304  zapZ(refsub);
305 
306  pTrajectoryWriter outtraj = otopts->createTrajectory(prefopts->prefix);
307  outtraj->setComments(header);
308 
309  bool first = true;
310  for (vector<uint>::iterator i = indices.begin(); i != indices.end(); ++i) {
311  traj->readFrame(*i);
312  traj->updateGroupCoords(model);
313 
314  GCoord original_center;
315  if (topts->no_ztrans)
316  original_center = applyto_sub.centroid();
317 
318  if (topts->xy_only) {
319  AtomicGroup flat_align = align_sub.copy();
320  zapZ(flat_align);
321  GMatrix M = flat_align.superposition(refsub);
322  M(2,2) = 1.0; // Fix zapped z-part (see above)
323  XForm W(M);
324  applyto_sub.applyTransform(W);
325  } else {
326  GMatrix M = align_sub.superposition(refsub);
327  XForm W(M);
328  applyto_sub.applyTransform(W);
329  }
330 
331  if (topts->no_ztrans) {
332  GCoord new_center = applyto_sub.centroid();
333  double dz = original_center.z() - new_center.z();
334  GCoord shift(0.0, 0.0, dz);
335  applyto_sub.translate(shift);
336  }
337 
338  outtraj->writeFrame(applyto_sub);
339 
340  if (first) {
341  savePDB(prefopts->prefix + ".pdb", header, applyto_sub);
342  first = false;
343  }
344  }
345 
346  }
347 }
