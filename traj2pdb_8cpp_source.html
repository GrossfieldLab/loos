47 string fullHelpMessage(void) {
48  string msg =
49  "\n"
50  "SYNOPSIS\n"
51  "\tWrite frames of a trajectory into PDB files\n"
52  "\n"
53  "DESCRIPTION\n"
54  "\n"
55  "\tThis tool converts a trajectory into a series of separate PDB files, one for\n"
56  "each frame in the trajectory. The output name template is a printf-format string\n"
57  "\n"
58  "EXAMPLES\n"
59  "\n"
60  "\ttraj2pdb model.psf simulation.dcd frame%03d.pdb\n"
61  "This creates frame000.pdb, frame001.pdb, frame002.pdb, etc\n"
62  "\n"
63  "\ttraj2pdb model.psf simulation.dcd sim%d.pdb\n"
64  "This creates sim0.pdb, sim1.pdb, ..., sim10.pdb, sim11.pdb ...\n"
65  "\n"
66  "NOTES\n"
67  "\tThere is no facility for extracting ranges of frames. Use subsetter to pre-process\n"
68  "the trajectory, then use traj2pdb to convert to PDB files.\n"
69  "\n";
70 
71  return(msg);
72 }

76 int main(int argc, char *argv[]) {
77  if (argc != 4) {
78  cerr << "Usage - traj2pdb model trajectory output-name-template\n";
79  cerr << fullHelpMessage();
80  }
81 
82  AtomicGroup model = createSystem(argv[1]);
83  pTraj traj = createTrajectory(argv[2], model);
84  string pdb_core = string(argv[3]);
85 
86  uint n = traj->nframes();
87  PDB pdb = PDB::fromAtomicGroup(model);
88  pdb.remarks().add(invocationHeader(argc, argv));
89 
90  cout << boost::format("There are %d atoms and %d frames.\n") % model.size() % n;
91 
92  cout << "Processing - ";
93  cout.flush();
94 
95  for (uint i=0; i<n; ++i) {
96  if (i % 250 == 0) {
97  cout << '.';
98  cout.flush();
99  }
100  traj->readFrame(i);
101  traj->updateGroupCoords(model);
102 
103  ostringstream s;
104  s << boost::format(pdb_core) % i;
105 
106  ofstream pdbout(s.str().c_str());
107  if (pdbout.fail())
108  {
109  cerr << "Error writing file "
110  << s.str()
111  << ". Exiting"
112  << endl;
113  exit(-1);
114  }
115 
116  pdbout << pdb;
117  pdbout.close();
118  }
119 
120  cout << " done\n";
121 }
