48 string fullHelpMessage(void)
49  {
50  string s =
51  "\n"
52  "SYNOPSIS\n"
53  "\n"
54  "\tCompute the radial distribution function for two selections of atoms\n"
55  "\n"
56  "DESCRIPTION\n"
57  "\n"
58  "\tThis tool computes the radial distribution function for two selections\n"
59  "of atoms, treating them as individual atoms rather than groups. This is\n"
60  "in contrast to the tool rdf, which treats them as groups.\n"
61  "\n"
62  "The output columns have the following meaning:\n"
63  " 1: distance\n"
64  " 2: normalized RDF\n"
65  " 3: cumulative distribution function of selection-2 atoms around \n"
66  " selection-1 atoms\n"
67  " 4: cumulative distribution function of selection-1 atoms around \n"
68  " selection-2 atoms\n"
69  "\n"
70  "EXAMPLE\n"
71  "\n"
72  " atomic-rdf model traj 'name =~ \"OP[1-4]\"' 'name =~ \"OH2\" && \\\n"
73  " resname == \"TIP3\"' 0 20 40\n"
74  "will compute the radial distribution function for phosphate oxygens and\n"
75  "water oxygens, treating each phosphate oxygen independently. Using the \n"
76  "same selections with the rdf tool would likely group the 4 phosphate \n"
77  "oxygens from each lipid into one unit and use their center of mass.\n"
78  "\n"
79  "As with the other rdf tools (rdf, xy_rdf), histogram-min, histogram-max,\n"
80  "and histogram-bins control the range over which the rdf is computed, and\n"
81  "the number of bins used, in this case from 0 to 20 Angstroms, with 0.5\n"
82  "angstrom bins.\n";
83  return(s);
84  }

86 int main (int argc, char *argv[])
87 {
88 
89 // Build options
90 opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
91 opts::TrajectoryWithFrameIndices* tropts = new opts::TrajectoryWithFrameIndices;
92 opts::RequiredArguments* ropts = new opts::RequiredArguments;
93 
94 // These are required command-line arguments (non-optional options)
95 ropts->addArgument("selection1", "selection1");
96 ropts->addArgument("selection2", "selection2");
97 ropts->addArgument("min", "min radius");
98 ropts->addArgument("max", "max radius");
99 ropts->addArgument("num_bins", "number of bins");
100 
101 opts::AggregateOptions options;
102 options.add(bopts).add(tropts).add(ropts);
103 if (!options.parse(argc, argv))
104  exit(-1);
105 
106 
107 // Print the command line arguments
108 cout << "# " << invocationHeader(argc, argv) << endl;
109 
110 // copy the command line variables to real variable names
111 // Create the system and read the trajectory file
112 AtomicGroup system = tropts->model;
113 pTraj traj = tropts->trajectory;
114 if (!(system.isPeriodic() || traj->hasPeriodicBox()))
115  {
116  cerr << "Error- Either the model or the trajectory must have periodic box information.\n";
117  exit(-1);
118  }
119 
120 // Extract our required command-line arguments
121 string selection1 = ropts->value("selection1"); // String describing the first selection
122 string selection2 = ropts->value("selection2"); // String describing the second selection
123 double hist_min = parseStringAs<double>(ropts->value("min")); // Lower edge of the histogram
124 double hist_max = parseStringAs<double>(ropts->value("max")); // Upper edge of the histogram
125 int num_bins = parseStringAs<double>(ropts->value("num_bins")); // Number of bins in the histogram
126 
127 
128 double bin_width = (hist_max - hist_min)/num_bins;
129 
130 
131 // Set up the selector to define group1 atoms
132 AtomicGroup group1 = selectAtoms(system, selection1);
133 if (group1.empty())
134  {
135  cerr << "Error- no atoms selected by '" << selection1 << "'\n";
136  exit(-1);
137  }
138 
139 // Set up the selector to define group2 atoms
140 AtomicGroup group2 = selectAtoms(system, selection2);
141 if (group2.empty())
142  {
143  cerr << "Error- no atoms selected by '" << selection2 << "'\n";
144  exit(-1);
145  }
146 
147 // Create the histogram and zero it out
148 vector<double> hist;
149 hist.reserve(num_bins);
150 hist.insert(hist.begin(), num_bins, 0.0);
151 
152 double min2 = hist_min*hist_min;
153 double max2 = hist_max*hist_max;
154 
155 // loop over the frames of the trajectory
156 vector<uint> framelist = tropts->frameList();
157 uint framecnt = framelist.size();
158 double volume = 0.0;
159 unsigned long unique_pairs=0;
160 for (uint index = 0; index<framecnt; ++index)
161  {
162  traj->readFrame(framelist[index]);
163 
164  // update coordinates and periodic box
165  traj->updateGroupCoords(system);
166  GCoord box = system.periodicBox();
167  volume += box.x() * box.y() * box.z();
168 
169  unique_pairs = 0;
170  // compute the distribution of g2 around g1
171  for (uint j = 0; j < group1.size(); j++)
172  {
173  pAtom a1 = group1[j];
174  GCoord p1 = a1->coords();
175  for (uint k = 0; k < group2.size(); k++)
176  {
177  pAtom a2 = group2[k];
178  // skip "self" pairs
179  if (a1 == a2)
180  {
181  continue;
182  }
183  unique_pairs++;
184  GCoord p2 = a2->coords();
185  // Compute the distance squared, taking periodicity into account
186  double d2 = p1.distance2(p2, box);
187  if ( (d2 < max2) && (d2 > min2) )
188  {
189  double d = sqrt(d2);
190  int bin = int((d-hist_min)/bin_width);
191  hist[bin]++;
192  }
193  }
194  }
195  }
196 
197 volume /= framecnt;
198 
199 
200 
201 double expected = framecnt * unique_pairs / volume;
202 double cum1 = 0.0;
203 double cum2 = 0.0;
204 
205 // Output the results
206 cout << "# Dist\tRDF\tCumAround1\tCumAround2" << endl;
207 for (int i = 0; i < num_bins; i++)
208  {
209  double d = bin_width*(i + 0.5);
210 
211  double d_inner = bin_width*i;
212  double d_outer = d_inner + bin_width;
213  double norm = 4.0/3.0 * M_PI*(d_outer*d_outer*d_outer
214  - d_inner*d_inner*d_inner);
215 
216  double total = hist[i]/ (norm*expected);
217  cum1 += hist[i] / (framecnt*group1.size());
218  cum2 += hist[i] / (framecnt*group2.size());
219 
220  cout << d << "\t" << total << "\t"
221  << cum1 << "\t" << cum2 << endl;
222 
223  }
224 }
