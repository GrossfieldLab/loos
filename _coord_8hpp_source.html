353 T uvdot(const Coord<T>& rhs) const {
354 // define quantities to accumulate with first step of loop.
355 T s = v[0] * rhs.v[0];
356 T d = v[0] * v[0];
357 T drhs = rhs.v[0] * rhs.v[0];
358
359 for (uint i = 1; i < MAXCOORD; ++i){
360 // dot product
361 s += v[i] * rhs.v[i];
362 // length2 of this
363 d += v[i] * v[i];
364 // length2 of rhs
365 drhs += rhs.v[i] * rhs.v[i];
366 }
367 return(s / sqrt(d * drhs));
368 }

378 Coord<T> cross(const Coord<T>& rhs) const {
379 Coord<T> res;
380 res.v[X] = v[Y] * rhs.v[Z] - v[Z] * rhs.v[Y];
381 res.v[Y] = v[Z] * rhs.v[X] - v[X] * rhs.v[Z];
382 res.v[Z] = v[X] * rhs.v[Y] - v[Y] * rhs.v[X];
383
384 return(res);

422 void reimage(const Coord<T>& box) {
423 for (uint i=0; i<MAXCOORD; ++i) {
424 int n = (int)(fabs(v[i]) / box.v[i] + 0.5);
425 v[i] = (v[i] >= 0) ? v[i] - n*(box.v[i]) : v[i] + n*(box.v[i]);
426 }
427 }

479 void random(void) {
480 base_generator_type& rng = rng_singleton();
481 boost::uniform_on_sphere<> uni(3);
482 boost::variate_generator<base_generator_type&,
483 boost::uniform_on_sphere<> > func(rng, uni);
484 std::vector<double> vec = func();
