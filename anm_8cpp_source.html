131  "ANM also supports construction of spring connections based on\n"
132  "pseudo-connectivity. This allows beads neighboring in sequence\n"

183 loos::Math::Matrix<int> buildConnectivity(const AtomicGroup& model) {
184  uint n = model.size();
185  loos::Math::Matrix<int> M(n, n);
186 
187  for (uint j=0; j<n-1; ++j)
188  for (uint i=j; i<n; ++i)
189  if (i == j)
190  M(j, i) = 1;
191  else {
192  M(j, i) = model[j]->isBoundTo(model[i]);
193  M(i, j) = M(j, i);
194  }
195 
196  return(M);
197 }

238  // Determine which kind of scaling to apply to the Hessian...
239  vector<SpringFunction*> springs;
240  SpringFunction* spring = 0;
241  spring = springFactory(spring_desc);
242  springs.push_back(spring);
243 
244  vector<SuperBlock*> blocks;
245  SuperBlock* blocker = new SuperBlock(spring, subset);
246  blocks.push_back(blocker);
247 
248 
249  // Handle Decoration (if necessary)
250  if (!bound_spring_desc.empty()) {
251  if (! model.hasBonds()) {
252  cerr << "Error- cannot use bound springs unless the model has connectivity\n";
253  exit(-10);
254  }
255  loos::Math::Matrix<int> M = buildConnectivity(subset);
256  SpringFunction* bound_spring = springFactory(bound_spring_desc);
257  springs.push_back(bound_spring);
258 
259  BoundSuperBlock* decorator = new BoundSuperBlock(blocker, bound_spring, M);
260  blocks.push_back(decorator);
261 
262  blocker = decorator;
263  }
264 
265

277  writeAsciiMatrix(prefix + "_s.asc", anm.eigenvalues(), header, false);
278 
279  writeAsciiMatrix(prefix + "_Hi.asc", anm.inverseHessian(), header, false);
280 
281  for (vector<SuperBlock*>::iterator i = blocks.begin(); i != blocks.end(); ++i)
282  delete *i;
283  for (vector<SpringFunction*>::iterator i = springs.begin(); i != springs.end(); ++i)
284  delete *i;
285
