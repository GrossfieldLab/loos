3 
4  A general purpose tool for subsetting a trajectory. This tool can
5  be used to extract specific atoms from a trajectory or specific
6  frames. It can also be used to add periodic box information (or
7  correct it) to a trajectory. It can also be used to concatenate
8  trajectories together (optionally extracting a subset of the
9  concatenated trajectory). Finally, you can center the output so
10  that the centroid of the selection is the origin. Note that the
11  selection used for centering comes from the specified subset...
12 
13  The output is always in DCD format.

95 struct ScanfFmt {
96  ScanfFmt(const string& s) : fmt(s) { }
97 
98  uint operator()(const string& s) const {
99  uint d;
100  if (sscanf(s.c_str(), fmt.c_str(), &d) != 1) {
101  cerr << boost::format("Bad conversion of '%s' using format '%s'\n") % s % fmt;
102  exit(-20);
103  }
104 
105  return(d);
106  }
107 
108  string fmt;
109 };

113 struct RegexFmt {
114  RegexFmt(const string& s) : fmt(s) {
115  regexp = boost::regex(s, boost::regex::perl);
116  }
117 
118  uint operator()(const string& s) const {
119  boost::smatch what;
120  if (boost::regex_search(s, what, regexp)) {
121  for (uint i=0; i<what.size(); ++i) {
122  string submatch(what.str(i));
123  char *p;
124  if (submatch.empty()) // Should never happen?
125  continue;
126 
127  uint val = strtoul(submatch.c_str(), &p, 10);
128  if (*p == '\0')
129  return(val);
130  }
131  }
132 
133  cerr << boost::format("Bad conversion of '%s' using regexp '%s'\n") % s % fmt;
134  exit(-20);
135  }
136 
137  string fmt;
138  boost::regex regexp;
139 };

160 // Given a vector of trajectory filenames, along with a functor for
161 // extracting the frame index from the filename, sorts it in numeric
162 // ascending order...
163 
164 template<class FmtOp>
165 vector<string> sortNamesByFormat(vector<string>& names, const FmtOp& op) {
166  uint n = names.size();
167  vector<SortDatum> bound(n);
168  for (uint i=0; i<n; ++i) {
169  bound[i] = SortDatum(names[i], op(names[i]));
170  }
171 
172  sort(bound.begin(), bound.end());
173 
174  vector<string> sorted(n);
175  for (uint i=0; i<n; ++i)
176  sorted[i] = bound[i].s;
177 
178  return(sorted);
179 }

192  "DESCRIPTION\n"
193  "\n"
194  "\tThis is a general-purpose tool (similar to catdcd from NAMD) that can be used\n"
195  "to convert trajectories to the DCD format, extract ranges of frames from a trajectory,\n"

199  "\tReimaging can be handled several different ways. The simplest is to turn on reimaging\n"
200  "with --reimage=normal. This reimages by molecule. In some cases, this is insufficient to\n"

202  "--reimage=aggressive. This employs a more aggressive reimaging that attempts to keep all parts of\n"
203  "a molecule together (the method used is similar to the --fix-imaging option). A similar\n"
204  "reimaging strategy is to use --reimage=zealous, a two-pass strategy where first normal\n"
205  "reimaging is applied, followed by aggressive. This can be helpful with some split GROMACS systems.\n"

220  "\n"

224  "\n"
225  "\tsubsetter -c 'segid == \"HEME\"' -s '!hydrogen' out model.pdb traj.dcd\n"
226  "This pulls all non-hydrogen atoms out of the trajectory and writes\n"
227  "them to out.dcd, centering so that the HEME segment is at the\n"
228  "origin.\n"
229  "\n"
230  "\tsubsetter -r 0:49,150:10:300 out model.pdb traj1.dcd traj2.dcd\n"

247  "\n"
248  "\tsubsetter -t xtc out model.pdb *.dcd\n"
249  "Writes out an XTC formatted trajectory to out.xtc and model to\n"
250  "out.pdb. Concatenates all DCD trajectories in the current\n"
251  "directory."
252  "\n"
253  "\tsubsetter --reimage=extreme --center='all' --postcenter='segid == \"POPC\" out.dcd model.psf *.dcd\n"

256  "\n"
257  "NOTES\n"
258  "\n"
259  "\t* sorting *\n"
260  "\tThe sorting option addresses a problem where you want to combine a\n"
261  "set of trajectories that have have a linearly increasing id\n"
262  "associated with them, i.e. \"traj.0.dcd\", \"traj.1.dcd\", etc. If\n"
263  "you give \"traj.*.dcd\" on the command-line, you will [most likely]\n"

271  " traj.20.dcd\n"
272  " ...\n"
273  "\n"
274  "\tGiving subsetter the \"--sort\" option causes subsetter to extract a\n"
275  "number from the trajectory filename and sort based on that\n"
276  "number. There are two ways you can tell subsetter how to extract\n"
277  "that number. The first is to use a scanf-style format string, the\n"

303  "SEE ALSO\n"

487 int main(int argc, char *argv[]) {
488  string hdr = invocationHeader(argc, argv);
489 
490  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
491  opts::BasicSelection* sopts = new opts::BasicSelection("all");
492  opts::OutputTrajectoryTypeOptions* otopts = new opts::OutputTrajectoryTypeOptions();
493  ToolOptions* topts = new ToolOptions;
494 
495  opts::AggregateOptions options;
496  options.add(bopts).add(sopts).add(otopts).add(topts);
497  if (!options.parse(argc, argv)) {
498  cerr << "Note- available model file formats (filename suffix) are:\n";
499  cerr << availableSystemFileTypes("\t");
500  cerr << "Note- available trajectory file formats (filename suffix) are:\n";
501  cerr << availableTrajectoryFileTypes("\t");
502  exit(-1);
503  }
504 
505 
506  verbose = bopts->verbosity;
507  if (verbose)
508  cout << "# " << hdr << endl;
509 
510  AtomicGroup model = createSystem(model_name);
511  selection = sopts->selection;
512  AtomicGroup subset = selectAtoms(model, selection);
513  if (subset.empty()) {
514  cerr << "Error- no atoms selected in subset\n";
515  exit(-10);
516  }
517 
518  AtomicGroup centered;
519  if (!center_selection.empty()) {
520  centered = selectAtoms(subset, center_selection);
521  if (centered.empty()) {
522  cerr << "Error- no atoms selected for centering\n";
523  exit(-10);
524  }
525  }
526 
527  AtomicGroup postcentered;
528  if (!post_center_selection.empty()) {
529  postcentered = selectAtoms(subset, post_center_selection);
530  if (postcentered.empty()) {
531  cerr << "Error- no atoms selected for post-centering\n";
532  exit(-10);
533  }
534  }
535 
536  MultiTrajectory mtraj(traj_names, model, skip, stride);
537  if (verbose)
538  showTrajectoryTable(mtraj);
539 
540  // Wrap since some LOOS tools will expect a pTraj rather than a traj...
541  pTraj ptraj(&mtraj, boost::lambda::_1);
542 
543  indices = assignTrajectoryFrames(ptraj, topts->range_spec, 0, 1);
544 
545  pTrajectoryWriter trajout = otopts->createTrajectory(out_name);
546  if (trajout->hasComments())
547  trajout->setComments(hdr);
548 
549  bool first = true; // Flag to pick off the first frame for a
550  // reference structure
551 
552  // If reimaging, break out the subsets to iterate over...
553  vector<AtomicGroup> molecules;
554  if (reimage_mode != NONE ) {
555  if (!model.hasBonds()) {
556  cerr << "WARNING- the model has no connectivity. Assigning bonds based on distance.\n";
557  model.findBonds();
558  }
559 
560  if (model.hasBonds())
561  molecules = model.splitByMolecule();
562  else
563  molecules = model.splitByUniqueSegid();
564 
565  if (verbose)
566  cout << boost::format("Reimaging %d molecules\n") % molecules.size();
567  }
568 
569  // Setup for progress output...
570  PercentProgressWithTime watcher;
571  ProgressCounter<PercentTrigger, EstimatingCounter> slayer(PercentTrigger(0.25), EstimatingCounter(indices.size()));
572  slayer.attach(&watcher);
573  if (verbose)
574  slayer.start();
575 
576  // Iterate over all requested global-frames...
577  for (vector<uint>::const_iterator vi = indices.begin(); vi != indices.end(); ++vi) {
578 
579  mtraj.readFrame(*vi);
580  mtraj.updateGroupCoords(model);
581 
582  // Handle Periodic boundary conditions...
583  if (box_override) {
584  if (first && subset.isPeriodic())
585  cerr << "WARNING - overriding existing periodic box.\n";
586  model.periodicBox(box);
587  }
588 
589  // Handle centering...
590  if (center_flag) {
591  GCoord c = centered.centroid();
592  model.translate(-c);
593  }
594 
595 
596  if (reimage_mode != NONE) {
597  if (reimage_mode == AGGRESSIVE || reimage_mode == ZEALOUS) {
598  if (reimage_mode == ZEALOUS) {
599  for (vGroup::iterator mol = molecules.begin(); mol != molecules.end(); ++mol)
600  mol->mergeImage();
601  }
602  GCoord centroid = centered[0]->coords();
603  model.translate(-centroid);
604  for (vGroup::iterator mol = molecules.begin(); mol != molecules.end(); ++mol)
605  mol->reimage();
606 
607  for (uint i=0; i<2; ++i) {
608  centroid = centered.centroid();
609  model.translate(-centroid);
610  for (vGroup::iterator mol = molecules.begin(); mol != molecules.end(); ++mol)
611  mol->reimage();
612  }
613 
614  } else if (reimage_mode == EXTREME) {
615 
616  for (vGroup::iterator mol = molecules.begin(); mol != molecules.end(); ++mol) {
617  uint midpoint = mol->size() / 2;
618  GCoord c = (*mol)[midpoint]->coords();
619  mol->translate(-c);
620  mol->reimageByAtom();
621  mol->translate(c);
622  }
623 
624  GCoord last_c = centered.centroid();
625  bool first = true;
626  uint si;
