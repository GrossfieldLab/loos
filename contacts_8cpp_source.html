34 string fullHelpMessage(void)
35  {
36  string s =
37  "\n"
38  "SYNOPSIS\n"
39  "\n"
40  "Count the number of contacts between the centers of mass of two sets\n"
41  "of selections.\n"
42  "\n"
43  "DESCRIPTION\n"
44  "\n"
45  "This tool counts the number of contacts between two selections.\n"
46  "Each selection is split by unique segment name, and the various \n"
47  "segments are treated separately, using their centers of mass. \n"
48  "\n"
49  "This tool provides a subset of the functionality supplied by rdf;\n"
50  "if you need splitting by something other than segment, you're better off\n"
51  "using rdf and looking at the cumulative columns, which have equivalent \n"
52  "information. The only advantage to using this tool is that it\n"
53  "avoids taking the square root in the distance calculation, so \n"
54  "it might be a little bit faster.\n"
55  "\n"
56  "EXAMPLE\n"
57  "\n"
58  "contacts model.pdb traj.dcd 'segname ==\"RHOD\"' 'segname =~\"^L[0-9]+\"' 18\n"
59  "\n"
60  "This command line reads model.pdb, loops over the trajectory traj.dcd, \n"
61  "and looks at 2 selections. The first is segment RHOD, which is \n"
62  "the protein rhodopsin, while the second is a set of lipid molecules \n"
63  "with segment names L1, L2, etc. It'll report the time series of the \n"
64  "number of lipids with centers of mass within 18 angstroms of the center \n"
65  "of mass of the protein. It will also report the same data normalized \n"
66  "by the number of groups in the first and second selection, respectively.\n"
67  "In this case, that means that since there's 1 protein, the second and \n"
68  "third columns will be the same, while the fourth column will be the \n"
69  "second column divided by the number of lipids selected.\n"
70  "\n"
71  ;
72  return(s);
73  }

83 int main (int argc, char *argv[])
84 {
85 if ( (argc >=2) )
86  {
87  if (string(argv[1]) == string("-h"))
88  {
89  Usage();
90  exit(-1);
91  }
92  else if (string(argv[1]) == string("--fullhelp"))
93  {
94  cerr << fullHelpMessage() << endl;
95  exit(-1);
96  }
97  }
98 if (argc < 6)
99  {
100  Usage();
101  exit(-1);
102  }
103 
104  cout << "# " << invocationHeader(argc, argv) << endl;
105 
106  // copy the command line variables to real variable names
107  string model_filename(argv[1]);
108  string traj_filename(argv[2]);
109  string selection1(argv[3]);
110  string selection2(argv[4]);
111  double max = strtod(argv[5], 0);
112  double max2 = max*max;
113 
114  AtomicGroup model = createSystem(model_filename);
115  pTraj traj = createTrajectory(traj_filename, model);
116 
117 
118  // The assumption here is that selection1 will specify a bunch of molecules,
119  // eg, all lipid headgroups of type foo. g1 will be the group containing all
120  // of those atoms. However, what we'll actually want to do is work with the
121  // individual molecules' centers of mass, so we'll split g1 into a vector of
122  // individual segids (corresponding to individual lipids) called group1
123  AtomicGroup g1 = selectAtoms(model, selection1);
124  vector<AtomicGroup> group1 = g1.splitByUniqueSegid();
125 
126  // g2 / group2 works the same way
127  AtomicGroup g2 = selectAtoms(model, selection2);
128  vector<AtomicGroup> group2 = g2.splitByUniqueSegid();
129 
130 
131 
132  cout << "#Frame\tPairs\tPerGroup1\tPerGroup2" << endl;
133 
134  // loop over the frames of the dcd file
135  int frame = 0;
136  while (traj->readFrame())
137  {
138  // get the new coordinates
139  traj->updateGroupCoords(model);
140  int count = 0;
141 
142  // compute the number of contacts between group1 center of mass
143  // and group2 center of mass
144  vector<AtomicGroup>::iterator first;
145  for (first=group1.begin(); first!=group1.end(); first++)
146  {
147  GCoord com1 = first->centerOfMass();
148 
149  vector<AtomicGroup>::iterator second;
150  for (second=group2.begin(); second!=group2.end(); second++)
151  {
152  // exclude self pairs
153  if (*first == *second)
154  continue;
155 
156  GCoord com2 = second->centerOfMass();
157 
158  double d2 = com1.distance2(com2, model.periodicBox());
159  if ( (d2 <= max2) )
160  {
161  count++;
162  }
163  }
164  }
165 
166  // Output the results
167  double per_g1_atom = (double)count / group1.size();
168  double per_g2_atom = (double)count / group2.size();
169  cout << frame << "\t"
170  << count << "\t"
171  << per_g1_atom << "\t"
172  << per_g2_atom << endl;
173 
174  frame++;
175  }
176 }
