172  template<typename T>
173  std::vector<T> parseRange(const std::string& text, const T endpoint = 0) {
174  T a;
175  T b;
176  T c;
177  char sep;
178  std::vector<T> indices;
179  std::istringstream is(text);
180 
181  is >> a;
182  if (is.eof()) {
183  indices.push_back(a);
184  return(indices);
185  }
186 
187  sep = is.get();
188  bool is_negative = false;
189 
190  if (is.peek() == '*') {
191  b = endpoint;
192  is.get();
193  } else {
194  if (is.peek() == '-') {
195  is_negative = true;
196  is.get();
197  }
198  is >> b;
199  if (is.fail() || sep != ':')
200  throw(ParseError("Could not parse range (1) " + text));
201  }
202 
203  sep = is.get();
204  if (is.eof()) {
205  c = 1;
206  if (is_negative)
207  b = -b;
208 
209  is_negative = a > b;
210 
211  } else {
212  if (sep != ':')
213  throw(ParseError("Could not parse range (2) " + text));
214  c = b;
215  if (is.peek() == '*') {
216  b = endpoint;
217  is.get();
218  } else {
219  is >> b;
220  if (is.fail())
221  throw(ParseError("Could not parse range (3) " + text));
222  }
223  }
224 
225  // Some input validation...
226  if (a > b && !is_negative)
227  throw(ParseError("You must use a negative step to count down: " + text));
228  if (a < b && is_negative)
229  throw(ParseError("You must use a postive step to count up: " + text));
230  if (c == 0)
231  throw(ParseError("Thou shalt only use non-zero step sizes: " + text));
232 
233  if (is_negative) {
234 
235  T i;
236  for (i=a; i > b; i -= c)
237  indices.push_back(i);
238  if (a < -a && b == 0) // If unsigned type, cannot use >= 0 as test
239  indices.push_back(0);
240  else if (i >= b)
241  indices.push_back(i);
242 
243  } else
244  for (T i=a; i <= b; i += c)
245  indices.push_back(i);
246 
247  return(indices);
248  }

261  template<typename T>
262  std::vector<T> parseRangeList(const std::string& text, const T endpoint = 0) {
263  std::vector<std::string> terms;
264  std::set<T> indices;
265  std::insert_iterator< std::set<T> > ii(indices, indices.begin());
266 
267  boost::split(terms, text, boost::is_any_of(","), boost::token_compress_on);
268  std::vector<std::string>::const_iterator ci;
269  for (ci = terms.begin(); ci != terms.end(); ci++) {
270  if (ci->empty())
271  continue;
272  std::vector<T> result = parseRange<T>(*ci, endpoint);
273  std::copy(result.begin(), result.end(), ii);
274  }
275  std::vector<T> results(indices.size());
276  std::copy(indices.begin(), indices.end(), results.begin());
277  return(results);
278  }

319  template<typename T>
320  T parseStringAs(const std::string& source, const uint pos =0, const uint nelem =0) {
321  T val(0);
322 
323  if (pos >= source.size()) {
324  std::stringstream msg;
325  msg << "Missing Field at position " << pos << std::endl;
326  msg << "> " << source << std::endl;
327  throw(ParseError(msg.str()));
328  }
329 
330  uint n = !nelem ? source.size() - pos : nelem;
331  if (pos + n > source.size())
332  n = source.size() - pos + 1;
333 
334 
335 
336  std::string element(source.substr(pos, n));
337  std::istringstream iss(element);
338  if (!(iss >> val)) {
339  std::stringstream msg;
340  msg << "PARSE ERROR\n" << source << std::endl;
341  for (uint i=0; i<pos; ++i)
342  msg << ' ';
343  msg << '^';
344  if (n > 1)
345  for (uint i=1; i<n; ++i)
346  msg << '^';
347  msg << std::endl;
348  throw(ParseError(msg.str()));
349  }
350 
351  return(val);
352  }
