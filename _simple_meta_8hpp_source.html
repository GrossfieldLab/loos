90 friend std::istream& operator>>(std::istream& is, SimpleMeta& m) {
91 std::string buf;
92
93 m.data_.clear();
94 while (true) {
95 int c = is.peek();
96 if (c != '#')
97 break;
98 std::getline(is, buf);
99 if (is.fail() || is.eof())
100 throw(std::runtime_error("Error while reading metadata"));
101 m.data_.push_back(m.stripper(buf));
102 }
103 return(is);
104 }
