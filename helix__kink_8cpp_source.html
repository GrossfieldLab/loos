48 string fullHelpMessage(void) {

67  "NOTES\n"
68  "\n"

77 int main(int argc, char *argv[]) {
78  string header = invocationHeader(argc, argv);
79 
80  if (argc != 5) {
81  cerr << "Usage- helix_kink selection-1 selection-2 model trajectory\n";
82  cerr << fullHelpMessage();
83  exit(-1);
84  }

86 
87  int k = 1;
88  string pre_sel(argv[k++]);
89  string post_sel(argv[k++]);
90 
91  AtomicGroup model = createSystem(argv[k++]);
92  pTraj traj = createTrajectory(argv[k++], model);
93 
94  AtomicGroup pre = selectAtoms(model, pre_sel);
95  AtomicGroup post = selectAtoms(model, post_sel);
96 
97  cout << "# " << header << endl;
98  cout << boost::format("#%6s %10s %10s %10s %10s %10s %10s %10s\n") % "t" % "angle" % "x_0" % "y_0" % "z_0" % "x_1" % "y_1" % "z_1";
99 
100  uint t = 0;
101  while (traj->readFrame()) {
102  traj->updateGroupCoords(model);
103  Axes preax = pre.principalAxes();
104  Axes postax = post.principalAxes();
105 
106  GCoord u = preax[0];
107  GCoord v = -postax[0];
108  double angle = acos(u * v / (u.length() * v.length()) ) * RAD2DEG;
109  if (angle > 90)
110  angle = 180 - angle;
111  cout << boost::format("%6d %10lf %10lf %10lf %10lf %10lf %10lf %10lf\n") % (t++) % angle % u[0] % u[1] % u[2] % v[0] % v[1] % v[2];
112  }
