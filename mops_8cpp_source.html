52 string fullHelpMessage(void) {
53  string msg =
54  "\n"
55  "SYNOPSIS\n"
56  "Mops calculates a molecular-order parameter for a selection\n"
57  "\n"
58  "DESCRIPTION\n"
59  "\tMops is used to calculate an order parameter using a whole molecule.\n"
60  "This is used in cases such as coarse-grained simulations where there are\n"
61  "no hydrogens. For each molecule in the selection, the principal axes are\n"
62  "found. The 2nd and 3rd axes are treated as faux-hydrogens and their angle\n"
63  "with the z-axis is used to calculate an order parameter (as in order_params).\n"
64  "\n"
65  "The order parameters can be written out as a time-series. If multiple trajectories\n"
66  "are given, then there will be extra spaces between each trajectory in the output.\n"
67  "Individual trajectories can be plotted with gnuplot by using the 'index' keyword.\n"
68  "\n"
69  "Use the --residues=1 option to force splitting the selection by residue rather than\n"
70  "connectivity or unique segid.\n"
71  "\n"
72  "EXAMPLES\n"
73  "\tmops 'resname == \"POPC\"' model.gro simulation.xtc >order.asc\n"
74  "This computes a molecular order parameter for all POPC residues.\n"
75  "\n"
76  "\tmops --skip=50 'segid == \"LIPID\"' namd.psf sim1.dcd sim2.dcd sim3.dcd sim4.dcd >order.asc\n"
77  "This computes a molecular order parameter for all molecules with a LIPID segid,\n"
78  "averaging over all 4 trajectories. The first 50 frames of each trajectory are\n"
79  "skipped.\n"
80  "\n"
81  "NOTES\n"
82  "\tThe entire selected molecule is used for the order parameter. You will\n"

163 double rowStats(const RealMatrix& M, const uint row, double *std) {
164  double avg = 0.0;
165  for (uint i=0; i<M.cols(); ++i)
166  avg += M(row, i);
167  avg /= M.cols();
168 
169  double var = 0.0;
170  for (uint i=0; i<M.cols(); ++i) {
171  double d = M(row, i) - avg;
172  var += d*d;
173  }
174  var /= (M.cols() - 1);
175 
176  *std = sqrt(var/M.cols());
177  return(avg);
178 }

191 // Note: hist is really an average binned on distance
192 void principalComponentsOrder(dTimeSeries& order_parameters,
193  const vGroup& residues) {
194 
195  for (vGroup::const_iterator i = residues.begin(); i != residues.end(); ++i) {
196  AtomicGroup residue = i->copy();
197  residue.centerAtOrigin();
198  residue.mergeImage();
199  vector<GCoord> axes = residue.principalAxes();
200  bool planar = false;
201 
202  if (abs(axes[3].z()) < minp) {
203  if (nplanar == 0) {
204  PDB pdb = PDB::fromAtomicGroup(residue);
205  cerr << progname << ": Warning- PCA magnitudes out of bounds " << axes[3] << endl;
206  cerr << pdb;
207  }
208  planar = true;
209  ++nplanar;
210  }
211 
212  double order1 = 0.5 - 1.5 * axes[1].z() * axes[1].z();
213  double order2 = 0.5 - 1.5 * axes[2].z() * axes[2].z();
214 
215  order_parameters.push_back(order1);
216  ++ntotal;
217  if (!planar) {
218  order_parameters.push_back(order2);
219  ++ntotal;
220  }
221  }
222 }

226 vGroup extractSelections(const AtomicGroup& model, const string& selection, const bool force_residues) {
227  AtomicGroup subset = selectAtoms(model, selection);
228 
229  if (subset.empty()) {
230  cerr << "Error- no atoms were selected.\n";
231  exit(EXIT_FAILURE);
232  }
233 
234  vGroup residues;
235  if (force_residues) {
236  cerr << progname << ": Forcing split by residue\n";
237  residues = subset.splitByResidue();
238  } else {
239  if (subset.hasBonds()) {
240  cerr << progname << ": Model has connectivity. Using this to split selection.\n";
241  residues = subset.splitByMolecule();
242  } else {
243  residues = subset.splitByUniqueSegid();
244  }
245  }
246 
247  if (!force_residues && residues[0].size() == subset.size()) {
248  cerr << progname << ": GROMACS model suspected... Splitting by residue.\n";
249  residues = subset.splitByResidue();
250  }
251 
252  cerr << boost::format("%s: Extracted %d molecules from selection.\n") % progname % residues.size();
253  return(residues);
254 }

277  vString traj_names = topts->traj_names;
278 
279  cout << "# " << hdr << endl;

282 
283  // Calculate size of matrix to use...
284  ulong n = calculateSize(model, traj_names);
285  n -= traj_names.size() * skip;

288  dTimeSeries order;

290  uint file = 0;
291 
292  for (vString::const_iterator i = traj_names.begin(); i != traj_names.end(); ++i) {
293  dTimeSeries suborder;
294 
295  pTraj traj = createTrajectory(*i, model);
296  if (skip != 0)
297  traj->readFrame(skip-1);
298 
299  uint t = skip;
300  while (traj->readFrame()) {
301  dTimeSeries frameorder;
302  traj->updateGroupCoords(model);
303  principalComponentsOrder(frameorder, subset);
304  copy(frameorder.begin(), frameorder.end(), back_inserter(suborder));
305  if (topts->timeseries)
306  cout << file << '\t' << t++ << '\t' << frameorder.average() << '\t' << '\t' << frameorder.stdev() << endl;
307  }
308  if (topts->timeseries)
309  cout << endl << endl;
310  order.push_back(suborder.average());
311  ++file;
312  }
313 
314 
315  if (topts->timeseries) {
316  cerr << boost::format("%s: Average MOP = %f\n") % progname % order.average();
317  cout << "# Avg = " << order.average() << endl;
318  cout << "# Std = " << order.stdev() << endl;
319  cerr << boost::format("%s: OB Data = %d out of %d (%.2f%%)\n") % progname % nplanar % ntotal % (nplanar * 100.0 / ntotal);
320  } else {
321  cout << "Avg = " << order.average() << endl;
322  cout << "Std = " << order.stdev() << endl;
323  cout << boost::format("OB Data = %d out of %d (%.2f%%)\n") % nplanar % ntotal % (nplanar * 100.0 / ntotal);
324  }
325  exit(EXIT_SUCCESS);
326 }
