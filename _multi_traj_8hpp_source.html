82 void addTrajectory(const std::string& filename) {
83 pTraj traj = createTrajectory(filename, _model);
84 _trajectories.push_back(traj);
85 if (traj->nframes() > _skip)

98 uint nframes(const uint i) const {
99 if (i >= _trajectories.size())
100 throw(LOOSError("Requesting trajectory size for non-existent trajectory in MultiTraj"));
101
102 if (_trajectories[i]->nframes() <= _skip)
103 return 0;
104 return( (_trajectories[i]->nframes() - _skip + _stride - 1) / _stride );
105 }
