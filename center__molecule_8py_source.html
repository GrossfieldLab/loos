33 
34 # Create the model
35 model = createSystem(model_name)
36 
37 # Select the requested atoms
38 subset = selectAtoms(model, selection)
39 
40 # Compute centroid the old-fashioned way...
41 # Obviously, it is faster to call subset.centroid(),
42 # but we're demonstrating how else one could do it.
43 center = GCoord(0,0,0)
44 for atom in subset:
45  center = center + atom.coords()
46 center = center / len(subset)
47 
48 for atom in subset:
49  atom.coords(atom.coords() - center)
50 
51 # Convert to a PDB
52 pdb = PDB.fromAtomicGroup(model)
53 
54 # Add a REMARK to the PDB
55 pdb.remarks().add("Structure centered using '" + selection + "'")
56 
57 # Print it to stdout...
58 print(pdb)
