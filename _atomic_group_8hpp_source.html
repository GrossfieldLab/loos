65 struct unordered_pair_hash
66 {
67 std::size_t operator()(std::pair<T, T> const &pair) const
68 {
69 // order the elements in the pair
70 auto min_max = std::minmax(pair.first, pair.second);
71 // return either
72 return static_cast<size_t>(min_max.second) << sizeof(T) * CHAR_BIT | min_max.first;
73 }

973 // This function is to to remove code duplication in
974 // logisticContacts() and logisticContacts2D().
975 // Handle even and odd powers separately -- even can
976 // avoid the sqrt
977 // Sigh, this doesnt' seem to make it much faster...
978 double logisticFunc(const GCoord &cent, const GCoord &other, double radius, int sigma, const GCoord &box) const
979 {
980 double prod;
981 if (sigma % 2 == 0)
982 {
983 double distance2 = cent.distance2(other, box);
984 double ratio = distance2 / (radius * radius);
985 prod = ratio;
986 for (int j = 0; j < (sigma / 2) - 1; ++j)
987 {
988 prod *= ratio;
989 }
990 }
991 else
992 {
993 double distance = cent.distance(other, box);
994 double ratio = distance / radius;
995 prod = ratio;
996 for (int j = 0; j < sigma - 1; ++j)
997 {
998 prod *= ratio;
999 }
1000 }
1001 double sum = 1. / (1. + prod);
1002 return (sum);
1003 }

1005 // Find all atoms in the current group that are within dist
1006 // angstroms of any atom in the passed group. The distance
1007 // calculation is determined by the passed functor so that the
1008 // same code can be used for both periodic and non-periodic
1009 // coordinates.
1010
1011 template <typename DistanceCalc>
1012 AtomicGroup within_private(const double dist, AtomicGroup &grp, const DistanceCalc &distance_functor) const
1013 {
1014
1015 AtomicGroup res;
1016 res.box = box;
1017
1018 double dist2 = dist * dist;
1019 std::vector<uint> indices;
1020
1021 for (uint j = 0; j < size(); j++)
1022 {
1023 GCoord c = atoms[j]->coords();
1024 for (uint i = 0; i < grp.size(); i++)
1025 {
1026 if (distance_functor(c, grp.atoms[i]->coords()) <= dist2)
1027 {
1028 indices.push_back(j);
1029 break;
1030 }
1031 }
1032 }
1033
1034 if (indices.size() == 0)
1035 return (res);
1036
1037 for (std::vector<uint>::const_iterator ci = indices.begin(); ci != indices.end(); ++ci)
1038 res.addAtom(atoms[*ci]);
1039
1040 return (res);
1041 }

1043 template <typename DistanceCalc>
1044 bool contactwith_private(const double dist, const AtomicGroup &grp, const uint min_contacts, const DistanceCalc &distance_function) const
1045 {
1046 double dist2 = dist * dist;
1047 uint ncontacts = 0;
1048
1049 for (uint j = 0; j < size(); ++j)
1050 {
1051 GCoord c = atoms[j]->coords();
1052 for (uint i = 0; i < grp.size(); ++i)
1053 if (distance_function(c, grp.atoms[i]->coords()) <= dist2)
1054 if (++ncontacts >= min_contacts)
1055 return (true);
1056 }
1057 return (false);
1058 }

1064 template <typename DistanceCalc>
1065 void findBondsImpl(const double dist, const DistanceCalc &distance_function)
1066 {
1067 iterator ij;
1068 double dist2 = dist * dist;
1069 double current_dist2;
1070
1071 for (ij = begin(); ij != end() - 1; ++ij)
1072 {
1073 iterator ii;
1074 GCoord u = (*ij)->coords();
1075
1076 for (ii = ij + 1; ii != end(); ++ii)
1077 {
1078 current_dist2 = distance_function(u, (*ii)->coords());
1079 if (current_dist2 < dist2)
1080 {
1081 (*ij)->addBond(*ii);
1082 (*ii)->addBond(*ij);
1083 }
1084 }
1085 }
