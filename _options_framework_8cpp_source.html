105  BasicSplitBy::SplitType BasicSplitBy::methodToType(const std::string& name)
106  {
107  std::string canonical_name(name);
108  boost::to_upper(canonical_name);
109 
110  if (canonical_name == "NONE")
111  return(NONE);
112  else if (canonical_name == "MOL")
113  return(MOLECULE);
114  else if (canonical_name == "SEGID")
115  return(SEGID);
116  else if (canonical_name == "RES")
117  return(RESIDUE);
118 
119 
120  throw(OptionsError("Invalid split-by method in options"));
121  }

129  std::cerr << "Error- '" << split_method << "' is an invalid method for splitting a selection.\n";
130  return(false);
131  }
132  catch (...) {
133  throw;
134  }
135 
136 
137  return(true);
138  }

140  std::vector<AtomicGroup> BasicSplitBy::split(const AtomicGroup& grp)
141  {
142  std::vector<AtomicGroup> result;
143 
144  switch(split_type) {
145  case NONE:
146  result.push_back(grp);
147  break;
148 
149  case MOLECULE:
150  result = grp.splitByMolecule();
151  break;
152 
153  case SEGID:
154  result = grp.splitByUniqueSegid();
155  break;
156 
157  case RESIDUE:
158  result = grp.splitByResidue();
159  break;
160 
161  default:
162  throw(LOOSError("Unknown split method"));
163  }
164 
165  return(result);
166  }

475  std::string MultiTrajOptions::trajectoryTable() const {
476  std::ostringstream oss;
477  if (!frame_index_spec.empty())
478  oss << "# Note- composite frame range used was '" << frame_index_spec << "'\n";
479  oss << "# traj\tstart\tend\tfilename\n";
480  uint start_cnt = 0;
481  uint j = 0;
482  for (uint i=0; i<mtraj.size(); ++i) {
483  uint n = mtraj.nframes(i);
484  if (n == 0)
485  oss << boost::format("# Warning- '%s' was skipped due to insufficient frames\n") % mtraj[i]->filename();
486  else {
487  oss << boost::format("# %d\t%d\t%d\t%s\n")
488  % j
489  % start_cnt
490  % (start_cnt + n - 1)
491  % mtraj[i]->filename();
492  ++j;
493  }
494  start_cnt += n;
495  }
496 
497  return oss.str();
498  }
