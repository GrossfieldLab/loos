813  // Find all atoms in the current group that are within dist
814  // angstroms of any atom in the passed group. The distance
815  // calculation is determined by the passed functor so that the
816  // same code can be used for both periodic and non-periodic
817  // coordinates.
818 
819  template <typename DistanceCalc>
820  AtomicGroup within_private(const double dist, AtomicGroup& grp, const DistanceCalc& distance_functor) const {
821 
822  AtomicGroup res;
823  res.box = box;
824 
825  double dist2 = dist * dist;
826  std::vector<uint> indices;
827 
828  for (uint j=0; j<size(); j++) {
829  GCoord c = atoms[j]->coords();
830  for (uint i=0; i<grp.size(); i++) {
831  if (distance_functor(c, grp.atoms[i]->coords()) <= dist2) {
832  indices.push_back(j);
833  break;
834  }
835  }
836  }
837 
838  if (indices.size() == 0)
839  return(res);
840 
841  for (std::vector<uint>::const_iterator ci = indices.begin(); ci != indices.end(); ++ci)
842  res.addAtom(atoms[*ci]);
843 
844  return(res);
845  }

848  template<typename DistanceCalc>
849  bool contactwith_private(const double dist, const AtomicGroup& grp, const uint min_contacts, const DistanceCalc& distance_function) const {
850  double dist2 = dist * dist;
851  uint ncontacts = 0;
852 
853  for (uint j = 0; j<size(); ++j) {
854  GCoord c = atoms[j]->coords();
855  for (uint i = 0; i<grp.size(); ++i)
856  if (distance_function(c, grp.atoms[i]->coords()) <= dist2)
857  if (++ncontacts >= min_contacts)
858  return(true);
859  }
860  return(false);
861  }

868  template<typename DistanceCalc>
869  void findBondsImpl(const double dist, const DistanceCalc& distance_function) {
870  iterator ij;
871  double dist2 = dist * dist;
872  double current_dist2;
873 
874  for (ij = begin(); ij != end() - 1; ++ij) {
875  iterator ii;
876  GCoord u = (*ij)->coords();
877 
878  for (ii = ij + 1; ii != end(); ++ii) {
879  current_dist2 = distance_function(u, (*ii)->coords());
880  if (current_dist2 < dist2) {
881  (*ij)->addBond(*ii);
882  (*ii)->addBond(*ij);
883  }
884  }
885  }
886  }
