70 string fullHelpMessage(void) {
71  string msg =
72  "SYNOPSIS\n"
73  "\n"
74  "\tIdentify the blob closest to a user-specified criterion\n"
75  "\n"
76  "DESCRIPTION\n"
77  "\n"
78  "\tpick_blobs finds the blob closest to a user input. Several methods\n"
79  "of input are supported. For instance, given a pdb and selection string\n"
80  "the blob closest to the selection will be returned. Additionally, a blob\n"
81  "may be selected using its ID (see blobid). A point within the grid may\n"
82  "also be used. A range of distances and the largest blob within the range\n"
83  "are alternate criteria.\n"
84  "\n"
85  "The input is an integer grid (from blobid), and the output is another integer-grid\n"
86  "that can be used to mask a density grid.\n"
87  "\n"
88  "EXAMPLES\n"
89  "\tblobid --threshold 1 <foo.grid >foo_id.grid\n"
90  "\tpick_blob --model foo.pdb --selection 'resid==65' < foo_id.grid > foo_picked.grid\n"
91  "This example first segments the density at 1.0, and then picks the blob closest to\n"
92  "any atom in residue 65 in the model.\n"
93  "\n"
94  "\tpick_blob --point '(13,7,3)' <foo.grid >foo_picked.grid\n"
95  "This example picks the blob nearest coordinates (13,7,3) in real-space (i.e.\n"
96  "Angstroms).\n"
97  "\n"
98  "\tpick_blob --model foo.pdb --selection 'resid==65' --range 15 <foo_id.grid >foo_picked.grid\n"
99  "This example finds ALL blobs that are within 15 Angstroms of any atom in residue 65.\n"
100  "\n"
101  "\tpick_blob --model foo.pdb --selection 'resid==64' --range 15 --largest 1 <foo_id.grid >foo_picked.grid\n"
102  "This example is as above, except that only the largest blob within 15 Angstroms is picked,\n"
103  "rather than ALL blobs within 15 Angstroms.\n"
104  "\n"
105  "SEE ALSO\n"
106  "\tblobid, gridmask\n";
107 
108  return(msg);
109 }

155 void zapGrid(DensityGrid<int>& grid, const vector<int>& vals) {
156  DensityGridpoint dims = grid.gridDims();
157 
158  for (int k=0; k<dims[2]; k++)
159  for (int j=0; j<dims[1]; j++)
160  for (int i=0; i<dims[0]; i++) {
161  DensityGridpoint point(i,j,k);
162  int val = grid(point);
163  vector<int>::const_iterator ci = find(vals.begin(), vals.end(), val);
164  if (ci == vals.end())
165  grid(point) = 0;
166  }
167 }

170 int maxBlobId(const DensityGrid<int>& grid) {
171  DensityGridpoint dims = grid.gridDims();
172  long k = dims[0] * dims[1] * dims[2];
173  int maxid = 0;
174 
175  for (long i=0; i<k; i++)
176  if (grid(i) > maxid)
177  maxid = grid(i);
178 
179  return(maxid);

183 vector<Blob> pickBlob(const DensityGrid<int>& grid, const vector<GCoord>& points) {
184  vector<DensityGridpoint> gridded;
185  vector<GCoord>::const_iterator ci;
186 
187  for (ci = points.begin(); ci != points.end(); ++ci)
188  gridded.push_back(grid.gridpoint(*ci));
189 
190  int maxid = maxBlobId(grid);
191 
192  if (debug >= 1)
193  cerr << boost::format("Found %d total blobs in grid.\n") % maxid;
194 
195  vector<Blob> blobs(maxid+1, Blob());
196 
197  DensityGridpoint dims = grid.gridDims();
198  for (int k=0; k<dims[2]; k++)
199  for (int j=0; j<dims[1]; j++)
200  for (int i=0; i<dims[0]; i++) {
201  DensityGridpoint point(i,j,k);
202  int id = grid(point);
203  if (!id)
204  continue;
205 
206  vector<DensityGridpoint>::iterator cj;
207  for (cj = gridded.begin(); cj != gridded.end(); ++cj) {
208  double d = point.distance2(*cj);
209  if (d < blobs[id].grid_dist) {
210  blobs[id].id = id;
211  blobs[id].grid_dist = d;
212  blobs[id].closest_point = point;
213  GCoord a = grid.gridToWorld(point);
214  GCoord b = grid.gridToWorld(*cj);
215  blobs[id].real_dist = a.distance2(b);
216  }
217  }
218  }
219 
220  if (debug > 1) {
221  cerr << "* DEBUG: Blob list dump *\n";
222  for (int i=0; i<=maxid; i++)
223  cerr << boost::format("\tid=%d, grid_dist=%12.8g, real_dist=%12.8g\n")
224  % blobs[i].id
225  % blobs[i].grid_dist
226  % blobs[i].real_dist;
227  }
228 
229  vector<Blob> res;
230  if (range == 0.0) {
231  double min = numeric_limits<double>::max();
232  int id = -1;
233  for (int i=1; i<=maxid; i++)
234  if (blobs[i].grid_dist < min) {
235  min = blobs[i].grid_dist;
236  id = i;
237  }
238 
239  if (id > 0)
240  res.push_back(blobs[id]);
241 
242  } else {
243  for (int i=1; i<=maxid; i++)
244  if (blobs[i].real_dist <= range)
245  res.push_back(blobs[i]);
246  }
247 
248  return(res);
249 }

254 int main(int argc, char *argv[]) {
255  string header = invocationHeader(argc, argv);
256 
257  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
258  ToolOptions* topts = new ToolOptions;
259 
260  opts::AggregateOptions options;
261  options.add(bopts).add(topts);
262  if (!options.parse(argc, argv))
263  exit(-1);
264 
265  vector<GCoord> points;
266  if (picked_ids.empty()) {
267  if (use_spot)
268  points.push_back(spot);
269  else {
270  AtomicGroup model = createSystem(model_name);
271  AtomicGroup subset = selectAtoms(model, selection);
272 
273  for (AtomicGroup::iterator i = subset.begin(); i != subset.end(); ++i)
274  points.push_back((*i)->coords());
275  }
276  }
277 
278  DensityGrid<int> grid;
279  cin >> grid;
280 
281  cerr << "Read in grid with dimensions " << grid.gridDims() << endl;
282 
283  GCoord delta = grid.gridDelta();
284  double voxel_volume = 1.0 / delta[0];
285  for (int i=1; i<3; i++)
286  voxel_volume *= (1.0 / delta[i]);
287 
288  if (picked_ids.empty()) {
289 
290  vector<Blob> picks = pickBlob(grid, points);
291 
292  if (picks.empty()) {
293  cerr << "Warning - no blobs picked\n";
294  exit(0);
295  }
296 
297  cerr << "Picked " << picks.size() << " blobs:\n";
298  vector<Blob>::const_iterator ci;
299  vector<int> ids;
300  for (ci = picks.begin(); ci != picks.end(); ++ci) {
301  cerr << boost::format("\tid=%d, dist=%12.8g\n") % ci->id % ci->real_dist;
302  ids.push_back(ci->id);
303  }
304 
305  if (!query) {
306  zapGrid(grid, ids);
307  cout << grid;
308  }
309 
310  } else if (!query) {
311  zapGrid(grid, picked_ids);
312  cout << grid;
313  }
314 
315  exit(0);
316 }
