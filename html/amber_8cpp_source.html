33  // Parse simple Fortran format specifications, extracted from a %FORMAT tag...
34  // Takes a string of expected format types (characters) that the extracted format
35  // is compared against. For example, to parse floats, expected could be any of "FEG".
36  // If the format type does not match (for example, say "%FORMAT (20I8)" when expected
37  // is "FEG"), then an error will be thrown.
38  //
39  // Format specs are converted to upper-case for validation
40 
41  Amber::FormatSpec Amber::parseFormat(const std::string& expected_types, const std::string& where) {
42 
43  reader.getNext();
44  char period;
45  FormatSpec fmt;
46 
47  // Verify line has a %FORMAT tag
48  if (reader.line().compare(0, 7, "%FORMAT") != 0)
49  throw(FileReadErrorWithLine(reader.name(), "Expected format for " + where, reader.lineNumber()));
50 
51  // Extract format spec between parens...
52  boost::char_separator<char> sep("()");
53  std::string input_line(reader.line()); // Ubuntu 16.04 apparently requires that the string in tokenizer
54  tokenizer tokens(input_line, sep); // be non-const
55  tokenizer::iterator toks = tokens.begin();
56 
57  ++toks;
58  std::istringstream iss(*toks);
59 
60  // try nXw.d first
61  if (! (iss >> fmt.repeat >> fmt.type >> fmt.width >> period >> fmt.precision) ) {
62  iss.clear();
63  iss.seekg(0);
64  // Now try nXw
65  if (! (iss >> fmt.repeat >> fmt.type >> fmt.width) ) {
66  iss.clear();
67  iss.seekg(0);
68  // Now try Xw
69  if (! (iss >> fmt.type >> fmt.width) ) {
70  iss.clear();
71  iss.seekg(0);
72  // And finally just try X
73  if (! (iss >> fmt.type) )
74  throw(FileReadErrorWithLine(reader.name(), "Cannot parse format for " + where, reader.lineNumber()));
75  }
76  }
77  }
78 
79  // Compare against expected types
80  std::string expected_types_UC = boost::to_upper_copy(expected_types);
81 
82  if (expected_types_UC.find_first_of(toupper(fmt.type)) == std::string::npos)
83  throw(FileReadErrorWithLine(reader.name(), "Invalid format type for " + where, reader.lineNumber()));
84 
85  return(fmt);
86 
87  }

90  void Amber::parseCharges() {
91  // Amber stores charges in units of electrons/18.2223
92  // See http://ambermd.org/FileFormats.php
93  const double conversion = 18.2223;
94  FormatSpec fmt = parseFormat("EFG", "charges");
95 
96  std::vector<double> charges = readBlock<double>(fmt.width);
97  if (charges.size() != atoms.size())
98  throw(FileReadErrorWithLine(reader.name(), "Error parsing charges from amber file", reader.lineNumber()));
99 
100  for (uint i=0; i<charges.size(); ++i)
101  atoms[i]->charge(charges[i]/conversion);
102  }

140  void Amber::assignResidues(void) {
141  if (!(residue_pointers.size() == nres && residue_labels.size() == nres))
142  throw(std::runtime_error("Unable to assign residues."));
143 
144  int curresid = 0;
145  uint i = 0;
146  std::string curresname;
147 
148  for (i=0; i<nres-1; i++) {
149  ++curresid;
150  curresname = residue_labels[i];
151  for (uint j = residue_pointers[i]; j<residue_pointers[i+1]; j++) {
152  atoms[j-1]->resid(curresid);
153  atoms[j-1]->resname(curresname);
154  }
155  }
156 
157  // Fix the end...
158  ++curresid;
159  curresname = residue_labels[i];
160  for (uint j = residue_pointers[i]-1; j<natoms; j++) {
161  atoms[j]->resid(curresid);
162  atoms[j]->resname(curresname);
163  }
164  }

169  void Amber::parseBonds(const uint n) {
170  FormatSpec fmt = parseFormat("I", "bonds");
171 
172 
173  std::vector<int> bond_list = readBlock<int>(fmt.width);
174 
175  if (bond_list.size() != 3*n)
176  throw(FileReadErrorWithLine(reader.name(), "Error parsing bonds in amber file", reader.lineNumber()));
177 
178  for (uint i=0; i<bond_list.size(); i += 3) {
179  if (bond_list[i] == bond_list[i+1])
180  continue;
181 
182  pAtom aatom = atoms[bond_list[i]/3];
183  pAtom batom = atoms[bond_list[i+1]/3];
184 
185  // Amber bond lists are not symmetric, so make sure we add both pairs...
186  if (!(aatom->isBoundTo(batom)))
187  aatom->addBond(batom);
188  if (!(batom->isBoundTo(aatom)))
189  batom->addBond(aatom);
190 
191  }
192 
193  }

196  void Amber::parsePointers() {
197  FormatSpec fmt = parseFormat("I", "pointers");
198 
199  std::vector<uint> pointers = readBlock<uint>(fmt.width);
200 
201  // Now build up the atomic-group...
202  if (atoms.size() != 0)
203  throw(std::logic_error("Internal error: trying to read in an amber parmtop into a non-empty group!"));
204 
205 
206  natoms = pointers[0];
207  nbonh = pointers[2];
208  mbona = pointers[3];
209  nres = pointers[11];
210 
211  for (uint i=0; i<natoms; i++) {
212  pAtom pa(new Atom);
213  pa->id(i+1);
214  pa->index(i);
215  atoms.push_back(pa);
216  }
217 
218  }

241  void Amber::parseBoxDimensions() {
242  FormatSpec fmt = parseFormat("E", "box dimensions");
243 
244  std::vector<double> dimensions = readBlock<double>(fmt.width);
245  const double epsilon = 1e-8;
246 
247  double angle = dimensions[0];
248  double diff = fabs(angle - 90.0);
249  if (diff > epsilon) {
250  std::cerr << "# Warning: prmtop specifies non-rectangular box: ignoring"
251  << std::endl;
252  return;
253  }
254 
255  GCoord box = GCoord(dimensions[1], dimensions[2], dimensions[3]);
256  periodicBox(box);
257  }

300  void Amber::read(std::istream& ifs) {
301  reader.stream(ifs);
302 
303  while (reader.getNext()) {
304 
305  boost::char_separator<char> sep(" \t");
306  std::string input_line(reader.line()); // See parseFormat() for explanation...required by Ubuntu 16
307  tokenizer tokens(input_line, sep);
308  tokenizer::iterator toks = tokens.begin();
309 
310  if (*toks != "%FLAG")
311  continue;
312 
313  ++toks;
314  if (*toks == "TITLE")
315  parseTitle();
316  else if (*toks == "POINTERS")
317  parsePointers();
318  else if (*toks == "ATOM_NAME")
319  parseAtomNames();
320  else if (*toks == "CHARGE")
321  parseCharges();
322  else if (*toks == "MASS")
323  parseMasses();
324  else if (*toks == "RESIDUE_LABEL")
325  parseResidueLabels();
326  else if (*toks == "RESIDUE_POINTER")
327  parseResiduePointers();
328  else if (*toks == "BONDS_INC_HYDROGEN")
329  parseBonds(nbonh);
330  else if (*toks == "BONDS_WITHOUT_HYDROGEN")
331  parseBonds(mbona);
332  else if (*toks == "BOX_DIMENSIONS")
333  parseBoxDimensions();
334  else if (*toks == "AMOEBA_REGULAR_BOND_NUM_LIST")
335  parseAmoebaRegularBondNumList();
336  else if (*toks == "AMOEBA_REGULAR_BOND_LIST")
337  parseAmoebaRegularBondList(_amoeba_regular_bond_num_list);
338 
339  }
340 
341  assignResidues();
342  deduceAtomicNumberFromMass();
343  setGroupConnectivity();
344  }
