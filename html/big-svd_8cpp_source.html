157 RealMatrix extractCoordinates(pTraj& traj, AtomicGroup& grp, const vector<uint>& indices) {
158  uint m = grp.size() * 3;
159  uint n = indices.size();
160 
161  RealMatrix A(m, n);
162  vector<double> avg(m, 0.0);
163 
164  for (uint i=0; i<n; ++i) {
165  traj->readFrame(indices[i]);
166  traj->updateGroupCoords(grp);
167  for (uint j=0; j<static_cast<uint>(grp.size()); ++j) {
168  GCoord c = grp[j]->coords();
169  A(3*j, i) = c.x();
170  avg[3*j] += c.x();
171 
172  A(3*j+1, i) = c.y();
173  avg[3*j+1] += c.y();
174 
175  A(3*j+2, i) = c.z();
176  avg[3*j+2] += c.z();
177  }
178  }
179 
180  for (uint j=0; j<m; ++j)
181  avg[j] /= n;
182 
183  for (uint i=0; i<n; ++i)
184  for (uint j=0; j<m; ++j)
185  A(j, i) -= avg[j];
186 
187  return(A);
188 }

210 void normalizeRows(RealMatrix& A) {
211  for (uint j=0; j<A.rows(); ++j) {
212  double sum = 0.0;
213  for (uint i=0; i<A.cols(); ++i)
214  sum += A(j, i) * A(j, i);
215 
216  sum = sqrt(sum);
217  for (uint i=0; i<A.cols(); ++i)
218  A(j, i) /= sum;
219  }
220 }

223 int main(int argc, char *argv[]) {
224 
225  string hdr = invocationHeader(argc, argv);
226 
227  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
228  opts::BasicSelection* sopts = new opts::BasicSelection("name == 'CA'");
229  opts::OutputPrefix* popts = new opts::OutputPrefix;
230  opts::TrajectoryWithFrameIndices* tropts = new opts::TrajectoryWithFrameIndices;
231  ToolOptions* topts = new ToolOptions;
232 
233  opts::AggregateOptions options;
234  options.add(bopts).add(sopts).add(popts).add(tropts).add(topts);
235  if (!options.parse(argc, argv))
236  exit(-1);
237 
238  TrackStorage store;
239 
240  AtomicGroup model = tropts->model;
241  AtomicGroup subset = selectAtoms(model, sopts->selection);
242  pTraj traj = tropts->trajectory;
243 
244  string prefix = popts->prefix;
245  vector<uint> indices = tropts->frameList();
246 
247  writeMap(prefix + ".map", subset);
248 
249  // Build AA'
250 
251  RealMatrix A = extractCoordinates(traj, subset, indices);
252  cerr << boost::format("Coordinate matrix is %d x %d\n") % A.rows() % A.cols();
253  store.allocate(A.rows() * A.cols());
254  if (topts->write_source_matrix)
255  writeAsciiMatrix(prefix + "_A.asc", A, hdr);
256 
257 
258  store.allocate(A.rows() * A.rows());
259  cerr << "Multiplying transpose...\n";
260  RealMatrix C = MMMultiply(A, A, false, true);
261  cerr << "Done!\n";
262 
263  // Compute [U,D] = eig(C)
264 
265  char jobz = 'V';
266  char uplo = 'L';
267  f77int n = A.rows();
268  f77int lda = n;
269  float dummy;
270  RealMatrix W(n, 1);
271  f77int lwork = -1;
272  f77int info;
273 
274  cerr << "Calling ssyev to get work size...\n";
275 
276  ssyev_(&jobz, &uplo, &n, C.get(), &lda, W.get(), &dummy, &lwork, &info);
277  if (info != 0) {
278  cerr << boost::format("ssyev failed with info = %d\n") % info;
279  exit(-10);
280  }
281 
282  lwork = static_cast<f77int>(dummy);
283  store.allocate(lwork);
284  float *work = new float[lwork+1];
285 
286  cerr << "Calling ssyev for eigendecomp...\n";
287  ssyev_(&jobz, &uplo, &n, C.get(), &lda, W.get(), work, &lwork, &info);
288  if (info != 0) {
289  cerr << boost::format("ssyev failed with info = %d\n") % info;
290  exit(-10);
291  }
292  cerr << "Finished!\n";
293 
294  reverseColumns(C);
295  cerr << "Writing LSVs...";
296  writeAsciiMatrix(prefix + "_U.asc", C, hdr);
297  cerr << "done.\n";
298 
299  // D = sqrt(D); Scale eigenvalues...
300  for (uint j=0; j<W.rows(); ++j)
301  W[j] = W[j] < 0 ? 0.0 : sqrt(W[j]);
302 
303  reverseRows(W);
304  writeAsciiMatrix(prefix + "_s.asc", W, hdr);
305 
306  // Multiply eigenvectors by inverse eigenvalues
307  for (uint i=0; i<C.cols(); ++i) {
308  double konst = (W[i] > 0.0) ? (1.0/W[i]) : 0.0;
309 
310  for (uint j=0; j<C.rows(); ++j)
311  C(j, i) *= konst;
312  }
313 
314  W.reset();
315  store.free(W.rows() * W.cols());
316 
317  store.allocate(A.cols() * A.rows());
318  cerr << "Multiplying to get RSVs...\n";
319  RealMatrix Vt = MMMultiply(C, A, true, false);
320  cerr << "Done!\n";
321  C.reset();
322  A.reset();
323 
324  cerr << "Writing RSVs...";
325  if (topts->subset_rsv) {
326  RealMatrix Vts = submatrix(Vt, loos::Math::Range(0, topts->subset_rsv), loos::Math::Range(0, Vt.cols()));
327  Vt=Vts;
328  }
329 
330  writeAsciiMatrix(prefix + "_V.asc", Vt, hdr, true);
331  cerr << "done.\n";
332
