99 string fullHelpMessage(void) {
100  string msg =
101  "\n"
102  "SYNOPSIS\n"
103  "\n"
104  "Calculate the normalized transition between two structures.\n"
105  "\n"
106  "\n"
107  "DESCRIPTION\n"
108  "\n"
109  "This tool will calculate the number of unique contacts in\n"
110  "a trajectory in order to asses its transition from a source\n"
111  "structure to a sink structure. Contacts are defined using\n"
112  "two input structures. The set of contacts which differ between\n"
113  "the two input structures are then used in the calculation.\n"
114  "This tool explicitly uses the side-chain centriods for the\n"
115  "calculation. (Note: In this context the word \'unique\' is used\n"
116  "to describe those contacts present in only the source OR the\n"
117  "sink structure.)\n"
118  "\n"
119  "The result is a normalized count of the number of contacts\n"
120  "broken and formed for each frame of the trajectory.\n"
121  "The output is 4 columns:\n"
122  "\tFrame - trajectory frame number\n"
123  "\tContacts broken - Number of unique source contacts broken\n"
124  "\tContacts formed - Number of unique sink contacts formed\n"
125  "\tTransition - Normalized sum of broken and formed\n"
126  "\n"
127  "Optionally, a timeseries of each contact's state may be\n"
128  "written out. By default, these values are scaled by a \n"
129  "tanh function centered at the cutoff. This effectively\n"
130  "smooths the transition matrix so contacts near the cutoff\n"
131  "have less \"flicker\" between broken and formed. This can\n"
132  "be changed to a step function by using the option\n"
133  "\t \"--smoothed-transition=false\"\n"
134  "In that case 1's are written for formed contacts and 0's\n"
135  "for broken contacts. Additionally, a list of all changed\n"
136  "contacts is output for reference.\n"
137  "\n"
138  "\n"
139  "\n"
140  //
141  "EXAMPLE\n"
142  "transition_contacts --cutoff 8 --selection \'segid==\"PROT\"\' model.pdb traj.dcd source.pdb sink.pdb\n"
143  "\tHere we are calculating how far our simulation has\n"
144  "\ttransitioned away from the structure in source.pdb\n"
145  "\ttowards the structure in sink.pdb. The selection\n"
146  "\tspecifies the entire segid \"PROT\" is used in the\n"
147  "\tcalculation. In this example the same selection is\n"
148  "\tapplied to both the source and sink models as well.\n"
149  "\tAn 8 angstrom cutoff is used to define connectivity.\n"
150  "\t\n"
151  "\t\n"
152  "transition_contacts --sink_sel 'resid >=0 && resid <= 10' --source_sel 'resid >= 100 && resid <= 110' --cutoff 8 --selection \'segid==\"PROT\"\' model.pdb traj.dcd source.pdb sink.pdb\n"
153  "\tSame as the above command, but now the selections\n"
154  "\tfor the source and sink models are separately specified.\n"
155  "\tIMPORTANT: It is your responsibility to ensure that\n"
156  "\tthe atoms selected in all three models match. This\n"
157  "\ttool will run regardless, pushing residues onto a \n"
158  "\tvector. \n"
159  "\t\n"
160  "\t\n"
161  "transition_contacts --timeseries output-time --selection \'segid==\"PROT\"\' model.pdb traj.dcd source.pdb sink.pdb\n"
162  "\tSame options as the first example, but now we\n"
163  "\toutput the timeseries of contacts to the file\n"
164  "\toutput-time in addition to the standard output.\n"
165  "\t\n"
166  "\t\n"
167  "transition_contacts --include-heavy 1 --timeseries output-time --selection \'segid==\"PROT\"\' model.pdb traj.dcd source.pdb sink.pdb\n"
168  "\tSame as the example above, but now the calculation\n"
169  "\tincludes backbone atoms and hydrogen atoms. Where\n"
170  "\tin previous examples these were excluded from the\n"
171  "\tcalculation.\n"
172  "\t\n"
173  "\t\n"
174  "SEE ALSO\n"
175  "native_contacts -\n"
176  "This tool calculates the changes in contacts when a 2nd structure\n"
177  "is not available.\n"
178  "\t\n"
179  "\t\n"

186 int main (int argc, char *argv[]){
187  string hdr = invocationHeader(argc, argv);
188 
189  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
190  opts::BasicTrajectory* tropts = new opts::BasicTrajectory;
191  ToolOptions* topts = new ToolOptions;
192 
193  opts::AggregateOptions options;
194  options.add(bopts).add(tropts).add(topts);
195  if (!options.parse(argc, argv))
196  exit(-1);
197  cout << "# " << hdr << endl;
198 
199  AtomicGroup model = tropts->model;
200  AtomicGroup system = selectAtoms(model, topts->selection);
201  AtomicGroup init_model = createSystem(topts->source_model);
202  AtomicGroup init = selectAtoms(init_model, topts->source_sel);
203  AtomicGroup final_model = createSystem(topts->sink_model);
204  AtomicGroup final = selectAtoms(final_model, topts->sink_sel);
205 
206  double cutoff = topts->cutoff;
207  pTraj traj = tropts->trajectory;
208  bool smoothed_transition = topts->smoothed_transition;
209 
210  // Check to see if user requested to leave these atoms...
211  if (!topts->leave_heavy){
212  system.remove(system.select(BackboneSelector()));
213  system.remove(system.select(HydrogenSelector()));
214  init.remove(init.select(BackboneSelector()));
215  init.remove(init.select(HydrogenSelector()));
216  final.remove(final.select(BackboneSelector()));
217  final.remove(final.select(HydrogenSelector()));
218 
219  if (system.empty() || init.empty() || final.empty()) {
220  cerr << "Error: The source, sink, or system is empty after post-processing.\n";
221  cerr << " Check your selections or use --include-heavy=1\n";
222  exit(-1);
223  }
224 
225 
226  }else{
227  cerr << "WARNING: Leaving backbone and hydrogen atoms!\n";
228  }
229 
230 
231 
232  // Split into residues
233  vector<AtomicGroup> start_residues = init.splitByResidue();
234  vector<AtomicGroup> final_residues = final.splitByResidue();
235  vector<AtomicGroup> residues = system.splitByResidue();
236 
237  if ((start_residues.size() != final_residues.size()) || (residues.size() != final_residues.size())) {
238  cerr << boost::format("Error: The trajectory has %d residues, the source has %d residues, and sink has %d residues.\n")
239  % residues.size()
240  % start_residues.size()
241  % final_residues.size();
242  cerr << "\tThe source and sink selections must have the same number of residues.\n";
243  exit(-1);
244  }
245 
246  double cut2 = cutoff*cutoff;
247  string timeseries_outfile = topts->timeseries;
248  ofstream ofs;
249  if (!timeseries_outfile.empty()){
250  ofs.open(timeseries_outfile.c_str());
251  if (ofs.fail()) {
252  cerr << "Error- failed to open '" << timeseries_outfile << "' for output.\n";
253  exit(-11);
254  }
255  ofs << "# " << hdr << endl;
256  ofs << "# Changed contacts list:\n";
257  ofs << "# ------------------------------\n";
258  }
259 
260 
261  //######################################################################
262  //### Build the master list of unique contacts
263  //######################################################################
264  vector<AtomicGroup> reslist;
265  vector<pair<uint, uint> > formed_connection_list;
266  vector<pair<uint, uint> > broken_connection_list;
267 
268  for (uint j = 0; j < residues.size()-1; ++j) {
269  uint jk;
270  bool any_connection_flag = false;
271  GCoord cj = start_residues[j].centerOfMass();
272  for (uint i = j+1; i < residues.size(); ++i) {
273  GCoord ci = start_residues[i].centerOfMass();
274  GCoord start_diff = cj - ci;
275 
276  // If formed in starting structure...
277  if (start_diff.length2() <= cut2){
278  GCoord fci = final_residues[i].centerOfMass();
279  GCoord fcj = final_residues[j].centerOfMass();
280  GCoord final_diff = fcj - fci;
281  // ...and if broken in ending structure
282  if (final_diff.length2() > cut2){
283  // Check if residue[j]'s first connection
284  // (we only want [j] listed once)
285  if (!any_connection_flag){
286  jk = reslist.size();
287  reslist.push_back(residues[j]);
288  any_connection_flag = true;
289  }
290  uint ik = reslist.size();
291  reslist.push_back(residues[i]);
292 
293  pair<uint,uint> conn_ptr(jk, ik);
294  // Add these to the broken pair list
295  broken_connection_list.push_back(conn_ptr);
296  // Document the residues in the broken list
297  ofs << "# broken: " << final_residues[j].getAtom(0)->resid() << " "
298  << final_residues[i].getAtom(0)->resid() << endl;
299  }
300  }
301 
302  // If broken in starting structure...
303  if (start_diff.length2() > cut2){
304  GCoord fci = final_residues[i].centerOfMass();
305  GCoord fcj = final_residues[j].centerOfMass();
306  GCoord final_diff = fcj - fci;
307  // ...and if formed in ending structure
308  if (final_diff.length2() <= cut2){
309  if (!any_connection_flag){
310  jk = reslist.size();
311  reslist.push_back(residues[j]);
312  any_connection_flag = true;
313  }
314  uint ik = reslist.size();
315  reslist.push_back(residues[i]);
316 
317  pair<uint,uint> conn_ptr(jk, ik);
318  // Add to the formed pair master list
319  formed_connection_list.push_back(conn_ptr);
320  // Document the residues in the formed list
321  ofs << "# formed: " << final_residues[j].getAtom(0)->resid() << " "
322  << final_residues[i].getAtom(0)->resid() << endl;
323  }
324  }
325  }
326  }
327 
328 
329  //######################################################################
330  //### Set up params for the calculation
331  //######################################################################
332  // Tally of the number of contacts for normalization
333  uint total_formed_contacts = formed_connection_list.size();
334  uint total_broken_contacts = broken_connection_list.size();
335 
336  cout << "# Total differences: \n";
337  cout << "# Contacts broken: " << total_broken_contacts << endl;
338  cout << "# Contacts formed: " << total_formed_contacts << endl;
339  cout << "# Frame \t broken \t formed \t total \n";
340  cout << "#-------------------------------------------\n";
341 
342 
343  //######################################################################
344  //### Iterate over trajectory frames
345  //### Calc the number of unique contacts broken/formed
346  //######################################################################
347  int frame = tropts->skip;
348  while (traj->readFrame()){
349  traj->updateGroupCoords(system);
350  double number_broken = 0.0;
351  double number_formed = 0.0;
352 
353  // Build a GCoord vector of CoM's
354  // with indices corresponding to reslist
355  vector<GCoord> centers;
356  for (uint gi = 0; gi < reslist.size(); ++gi){
357  GCoord temp_com = reslist[gi].centerOfMass();
358  centers.push_back(temp_com);
359  }
360 
361  // Do the actual calculation
362  // For broken list
363  for (uint i = 0; i < broken_connection_list.size(); ++i){
364 
365  GCoord first_current = centers[broken_connection_list[i].first];
366  GCoord second_current = centers[broken_connection_list[i].second];
367  bool broken = first_current.distance2(second_current) > cut2;
368  double current_dist = sqrt(first_current.distance2(second_current));
369 
370  if (smoothed_transition){
371  double value = 0.5 * tanh(current_dist-cutoff)+0.5;
372  number_broken += value;
373  if (!timeseries_outfile.empty())
374  ofs << value << '\t';
375  }else{
376  if (broken)
377  ++number_broken;
378  if (!timeseries_outfile.empty())
379  ofs << (broken ? '0' : '1') << '\t';
380  }
381  }
382 
383  // For formed list
384  for (uint i = 0; i < formed_connection_list.size(); ++i){
385  GCoord first_current = centers[formed_connection_list[i].first];
386  GCoord second_current = centers[formed_connection_list[i].second];
387  bool formed = first_current.distance2(second_current) < cut2;
388  double current_dist = sqrt(first_current.distance2(second_current));
389 
390  if (smoothed_transition){
391  double value = 0.5 * tanh(current_dist-cutoff)+0.5;
392  number_formed += value;
393  if (!timeseries_outfile.empty())
394  ofs << value << '\t';
395  }else{
396  if (formed)
397  ++number_formed;
398  if (!timeseries_outfile.empty())
399  ofs << (formed ? '1' : '0') << '\t';
400  }
401  }
402 
403  if (!timeseries_outfile.empty())
404  ofs << endl;
405 
406 
407  // Format and output the data
408  float num_transition = (number_broken + number_formed) / (total_broken_contacts + total_formed_contacts);
409  float percent_broken = number_broken / total_broken_contacts;
410  float percent_formed = number_formed / total_formed_contacts;
411  cout << frame << "\t" << percent_broken << "\t" << percent_formed << "\t" << num_transition << endl;
412  frame++;
413  }
