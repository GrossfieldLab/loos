137  "\n"
138  "This tool performs a principal component analysis (PCA)\n"
139  "on the trajectory. This technique computes a new coordinate\n"
140  "system such that the largest concerted motions are on the 1st\n"
141  "axis (the 1st principal component). This effectively reduces\n"
142  "relevant dimensionality of the system by resolving the most\n"

150  "their contents follows. For these descriptions assume an SVD is\n"
151  "\t\t A = UsV*\n"
152  "where the matrix A contains the coordinates of the atoms for every\n"
153  "frame in the trajectory:\n"
154  "\toutput_s.asc - singular values (square roots of eigenvalues)\n"
155  "\toutput_U.asc - left singular vectors (lsv, direction of each PC)\n"

269 Matrix extractCoords(const AtomicGroup& subset, const vector<XForm>& xforms, pTraj traj, const vector<uint>& indices) {
270 
271  AtomicGroup avg = averageStructure(subset, xforms, traj, indices);
272  writeAverage(avg);
273 
274  uint natoms = subset.size();
275  AtomicGroup frame = subset.copy();
276  uint n = indices.size();
277  uint m = natoms * 3;
278  Matrix M(m, n);
279 
280  for (uint i=0; i<n; ++i) {
281  traj->readFrame(indices[i]);
282  traj->updateGroupCoords(frame);
283  frame.applyTransform(xforms[i]);
284 
285  for (uint j=0; j<natoms; j++) {
286  GCoord c = frame[j]->coords() - avg[j]->coords();
287  M(j*3,i) = c.x();
288  M(j*3+1,i) = c.y();
289  M(j*3+2,i) = c.z();
290  }
291  }
292 
293  return(M);
294 }

337 int main(int argc, char *argv[]) {
338  header = invocationHeader(argc, argv);
339  opts::BasicOptions* bhopts = new opts::BasicOptions(fullHelpMessage());
340  opts::OutputPrefix* popts = new opts::OutputPrefix;
341  opts::MultiTrajOptions* tropts = new opts::MultiTrajOptions;
342  ToolOptions* topts = new ToolOptions;
343 
344  opts::AggregateOptions options;
345  options.add(bhopts).add(popts).add(tropts).add(topts);
346  if (!options.parse(argc, argv))
347  exit(-1);
348 
349  if (bhopts->verbosity)
350  cout << tropts->trajectoryTable() << endl;
351 
352  prefix = popts->prefix;
353  AtomicGroup model = tropts->model;
354  pTraj ptraj = tropts->trajectory;
355  vector<uint> indices = tropts->frameList();
356 
357  AtomicGroup svdsub = selectAtoms(model, topts->svd_string);
358 
359  write_map(prefix + ".map", svdsub);
360 
361  vector<XForm> xforms;
362  if (topts->noalign) {
363  // Make noop xforms to prevent doing any alignment...
364  cerr << argv[0] << ": SKIPPING ALIGNMENT\n";
365  for (uint i=0; i<indices.size(); ++i)
366  xforms.push_back(XForm());
367  } else {
368  AtomicGroup alignsub = selectAtoms(model, topts->alignment_string);
369  cerr << argv[0] << ": Aligning...\n";
370  xforms = doAlign(alignsub, ptraj, indices, topts->alignment_tol); // Honors indices
371  }
372 
373  cerr << argv[0] << ": Extracting coordinates...\n";
374  Matrix A = extractCoords(svdsub, xforms, ptraj, indices); // Honors indices
375  f77int m = A.rows();
376  f77int n = A.cols();
377  f77int sn = m<n ? m : n;
378 
379 
380  if (topts->include_source)
381  writeAsciiMatrix(prefix + "_A.asc", A, header);
382 
383  double estimate = static_cast<double>(m)*m*sizeof(svdreal) + static_cast<double>(n)*n*sizeof(svdreal) + static_cast<double>(m)*n*sizeof(svdreal) + sn*sizeof(svdreal);
384  cerr << boost::format("%s: Allocating estimated %.3f GB for %d x %d SVD\n")
385  % argv[0]
386  % (estimate / gigabytes)
387  % m
388  % n;
389 
390  char jobu = 'A', jobvt = 'A';
391  f77int lda = m, ldu = m, ldvt = n, lwork= -1, info;
392  svdreal prework[10], *work;
393 
394  Matrix U(m,m);
395  Matrix S(sn,1);
396  Matrix Vt(n,n);
397 
398  // First, request the optimal size of the work array...
399  SVDFUNC(&jobu, &jobvt, &m, &n, A.get(), &lda, S.get(), U.get(), &ldu, Vt.get(), &ldvt, prework, &lwork, &info);
400  if (info != 0) {
401  cerr << "Error code from size request to dgesvd was " << info << endl;
402  exit(-2);
403  }
404 
405  lwork = (f77int)prework[0];
406  estimate += lwork * sizeof(svdreal);
407  cerr << argv[0] << ": SVD requests " << lwork << " extra space for a grand total of " << estimate / gigabytes << " GB\n";
408  work = new svdreal[lwork];
409 
410  cerr << argv[0] << ": Calculating SVD...\n";
411  Timer<WallTimer> timer;
412  timer.start();
413  SVDFUNC(&jobu, &jobvt, &m, &n, A.get(), &lda, S.get(), U.get(), &ldu, Vt.get(), &ldvt, work, &lwork, &info);
414  timer.stop();
415  cerr << argv[0] << ": Done! Calculation took " << timeAsString(timer.elapsed()) << endl;
416 
417  if (info > 0) {
418  cerr << "Convergence error in dgesvd\n";
419  exit(-3);
420  } else if (info < 0) {
421  cerr << "Error in " << info << "th argument to dgesvd\n";
422  exit(-4);
423  }
424 
425 
426  Math::Range orig(0,0);
427  Math::Range Usize(m,m);
428  Math::Range Ssize(sn,1);
429  Math::Range Vsize(sn,n);
430 
431  if (topts->terms) {
432  int terms = static_cast<int>(topts->terms);
433  if (terms > m || terms > sn || terms > n) {
434  cerr << "ERROR- The number of terms requested exceeds matrix dimensions.\n";
435  exit(-1);
436  }
437  Usize = Math::Range(m, terms);
438  Ssize = Math::Range(terms, 1);
439  Vsize = Math::Range(terms, n);
440  }
441 
442  cerr << argv[0] << ": Writing results...\n";
443  writeAsciiMatrix(prefix + "_U.asc", U, header, orig, Usize);
444  writeAsciiMatrix(prefix + "_s.asc", S, header, orig, Ssize);
445 
446  if (topts->splitv && tropts->mtraj.size() > 1) {
447  // Need to reconstruct what row-ranges correspond to the input trajectories...
448  uint a = 0;
449  uint curtraj = 0;
450  int terms = topts->terms ? static_cast<int>(topts->terms) : sn;
451 
452  for (uint i=0; i<n; ++i) {
453  MultiTrajectory::Location loc = tropts->mtraj.frameIndexToLocation(i);
454  if (loc.first != curtraj) {
455  writeMatrixChunk(popts, tropts, topts, Vt, Math::Range(0, a), Math::Range(terms, i), header, curtraj);
456  a = i;
457  curtraj = loc.first;
458  }
459  }
460 
461  writeMatrixChunk(popts, tropts, topts, Vt, Math::Range(0, a), Math::Range(terms, n), header, curtraj);
462 
463  } else
464  writeAsciiMatrix(prefix + "_V.asc", Vt, header, orig, Vsize, true);
465 
466  cerr << argv[0] << ": done!\n";
467 
468  delete[] work;
469 }
