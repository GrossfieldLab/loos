161 class BinnedStatistics {
162 public:
163  typedef vector< vector<double> > BinType;
164  typedef pair<double,double> BinStatsType;
165 
166 
167 
168  BinnedStatistics(const double minval, const double maxval, const uint nbins) :
169  _minval(minval),
170  _nbins(nbins),
171  _delta( (maxval - minval) / nbins ),
172  _obdata(0),
173  _npts(0),
174  _bins(BinType(nbins))
175  { }
176 
177  void accumulate(const double coord, const double val) {
178  uint bin = (coord - _minval) / _delta;
179  if (bin >= _nbins) {
180  ++_obdata;
181  return;
182  }
183 
184  ++_npts;
185  _bins[bin].push_back(val);
186  }
187 
188  BinStatsType statisticsForBin(const uint bin) const {
189  uint n = _bins[bin].size();
190  if (n < 3)
191  return(BinStatsType(0.0, 0.0));
192 
193  double mean = 0.0;
194  for (uint i=0; i<n; ++i)
195  mean += _bins[bin][i];
196  mean /= n;
197 
198  double var = 0.0;
199  for (uint i=0; i<n; ++i) {
200  double d = _bins[bin][i] - mean;
201  var += d*d;
202  }
203 
204  return( BinStatsType(mean, sqrt(var/(n-1))/sqrt(n)) );
205  }
206 
207  uint numberOfPointsForBin(const uint bin) const { return(_bins[bin].size()); }
208  ulong numberOfDataPoints() const { return(_npts); }
209 
210 
211  double binCoordinate(const uint bin) const {
212  return(bin * _delta + _minval + _delta/2.0);
213  }
214 
215  ulong numberOutOfBounds() const { return(_obdata); }
216  uint numberOfBins() const { return(_nbins); }
217 
218 private:
219  double _minval;
220  uint _nbins;
221  double _delta;
222  ulong _obdata, _npts;
223  BinType _bins;
224 };

233 double centroidDistance(const AtomicGroup& a, const AtomicGroup& b) {
234  GCoord ac = a.centroid();
235  ac.z() = 0.0;
236 
237  GCoord bc = b.centroid();
238  bc.z() = 0.0;
239  return(ac.distance(bc));

242 // Uses x,y-plane distance between centroids
243 double minDistanceToSet(const AtomicGroup& a, const vecGroup& set) {
244 
245  double mind = numeric_limits<double>::max();
246  for (vecGroup::const_iterator i = set.begin(); i != set.end(); ++i) {
247  double d = centroidDistance(a, *i);
248  if (d < mind)
249  mind = d;
250  }
251 
252  return(mind);
253 }

258 // Note: hist is really an average binned on distance
259 void principalComponentsOrder(BinnedStatistics& phist,
260  BinnedStatistics& hist,
261  const vecGroup& molecules,
262  const vecGroup& lipopeptides) {
263  vecDouble orders;
264 
265  for (vecGroup::const_iterator i = molecules.begin(); i != molecules.end(); ++i) {
266  vector<GCoord> axes = i->principalAxes();
267  bool planar = false;
268 
269  if (axes[3].z() < minp) {
270  if (nplanar == 0) {
271  PDB pdb = PDB::fromAtomicGroup(*i);
272  cerr << "Warning- PCA magnitudes out of bounds " << axes[3] << endl;
273  cerr << pdb;
274  }
275  planar = true;
276  ++nplanar;
277  }
278 
279 
280  double order1 = 0.5 - 1.5 * axes[1].z() * axes[1].z();
281  double order2 = 0.5 - 1.5 * axes[2].z() * axes[2].z();
282 
283  GCoord lipid_center = i->centroid();
284  lipid_center.z() = 0.0;
285 
286  for (vecGroup::const_iterator j = lipopeptides.begin(); j != lipopeptides.end(); ++j) {
287  GCoord c = j->centroid();
288  c.z() = 0.0;
289  double d = lipid_center.distance(c);
290  phist.accumulate(d, abs(axes[0].z()));
291 
292  hist.accumulate(d, order1);
293  if (!planar)
294  hist.accumulate(d, order2);
295  }
296 
297  }
298 }

301 vecGroup filterByLeaflet(const vecGroup& ensemble, const LeafletType leaflet = UPPER) {
302  vecGroup side;
303 
304  for (vecGroup::const_iterator i = ensemble.begin(); i != ensemble.end(); ++i) {
305  GCoord c = i->centroid();
306  if (leaflet == UPPER && c.z() > 0)
307  side.push_back(*i);
308  else if (leaflet == LOWER && c.z() < 0)
309  side.push_back(*i);
310  }
311 
312  return(side);
313 }

316 vecGroup extractSelections(const AtomicGroup& model, const string& selection, const bool force_residues) {
317  AtomicGroup subset = selectAtoms(model, selection);
318 
319  if (subset.empty()) {
320  cerr << "Error- no atoms were selected.\n";
321  exit(EXIT_FAILURE);
322  }
323 
324  vecGroup residues;
325  if (force_residues) {
326  cerr << progname << ": Forcing split by residue\n";
327  residues = subset.splitByResidue();
328  } else {
329  if (subset.hasBonds()) {
330  cerr << progname << ": Model has connectivity. Using this to split selection.\n";
331  residues = subset.splitByMolecule();
332  } else {
333  residues = subset.splitByUniqueSegid();
334  }
335  }
336 
337  if (!force_residues && residues[0].size() == subset.size()) {
338  cerr << progname << ": Either you are using a GROMACS model or you have one molecule in your selection\n";
339  cerr << progname << ": If you are using GROMACS, you will want to run again with the --residue=1 option\n";
340  }
341 
342  cerr << boost::format("%s: Extracted %d molecules from selection.\n") % progname % residues.size();
343  return(residues);
344 }

372  cerr << boost::format("Lipopeptide selection has %d atoms per molecule and %d molecules.\n") % lipopeps[0].size() % lipopeps.size();
373 
374  BinnedStatistics lipid_phist(0.0, rmax, nbins); // Track the dot product of the first PC with membrane
375  // normal (z-axis)
376 
377  BinnedStatistics lipid_hist(0.0, rmax, nbins); // Track the fake hydrogen order parameters...
378 
379 
380  for (vector<string>::const_iterator ci = topts->traj_names.begin(); ci != topts->traj_names.end(); ++ci) {
381  pTraj traj = createTrajectory(*ci, model);
382  cerr << boost::format("Processing %s ...") % *ci;
383  cerr.flush();
384  if (skip > 0)
385  traj->readFrame(skip-1);
386 
387  while (traj->readFrame()) {
388  traj->updateGroupCoords(model);
389 
390 
391  vecGroup lip_leaf = filterByLeaflet(lipopeps, UPPER);
392  if (!lip_leaf.empty()) {
393  vecGroup lipid_leaf = filterByLeaflet(membrane, UPPER);
394  principalComponentsOrder(lipid_phist, lipid_hist, lipid_leaf, lip_leaf);
395  }
396 
397  lip_leaf = filterByLeaflet(lipopeps, LOWER);
398  if (!lip_leaf.empty()) {
399  vecGroup lipid_leaf = filterByLeaflet(membrane, LOWER);
400  principalComponentsOrder(lipid_phist, lipid_hist, lipid_leaf, lip_leaf);
401  }
402  }
403 
404  cerr << " done\n";
405  }
